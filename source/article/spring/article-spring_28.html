# HelloController

package com.striveonger.demo.client.web.controller;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.cloud.client.ServiceInstance;
import org.springframework.cloud.client.discovery.DiscoveryClient;
import org.springframework.stereotype.Controller;
import org.springframework.ui.Model;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.client.RestTemplate;

import java.time.LocalDateTime;
import java.util.List;
import java.util.Map;

/**
 * @author Mr.Lee
 * @description:
 * @date 2021-03-31 5:47 PM
 */
@Controller
public class HelloController {
    @Autowired
    private DiscoveryClient client;
    @GetMapping("/")
    public String handleRequest(Model model) {
        //accessing provider-hello
        List<ServiceInstance> instances = client.getInstances("provider-hello");
        if (instances != null && instances.size() > 0) {
            ServiceInstance instance = instances.get(0);
            RestTemplate restTemplate = new RestTemplate();
            String url = instance.getUri() + "/hello";
            Map map = restTemplate.getForObject(url, Map.class);
            model.addAttribute("msg", map.get("message"));
            model.addAttribute("time", LocalDateTime.now());
        }
        return "hello-page";
    }
}
