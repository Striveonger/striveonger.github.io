Spring Batch 批处理框架的技术调研

Mr.Lee 2019-05-04 18:30:23 SpringBootSpringBatch

Spring Batch是批处理的框架 - 执行一系列作业。

# 1. 关于 Spring Batch 如何使用

在Spring Batch中，作业由许多步骤组成，每个步骤由一个READ-PROCESS-WRITE任务或Single Operation 任务（tasklet）组成。

  • JobLauncher 任务的启动器
  • Job 封装整个批处理的实体. 一个任务(Job)可以由多个步骤(Step)组成.
  • Step 是任务中的一个步骤, 它封装了批处理任务中独立的执行单元.
  • ItemReader 是一个读取数据的抽象类
  • ItemProcessor 是一个处理读到数据的抽象类
  • ItemWriter 是一个存数数据的抽象类
  • JobRepository 用于记录任务每个阶段的 JobExecution

import org.springframework.batch.core.JobExecution;
import org.springframework.batch.core.JobExecutionListener;
import org.springframework.batch.item.ExecutionContext;

import java.util.HashMap;
import java.util.Map;

/**
 * @description: 给Job设置全局变量
 * @author: Mr.Lee
 * @date: 2021-08-23 14:22
 */
public class JobGlobalVariable implements JobExecutionListener {

    private Map<String, Object> params;

    private JobGlobalVariable() {
        this.params = new HashMap<>();
    }

    public void setParams(Map<String, Object> params) {
        this.params = params;
    }

    @Override
    public void beforeJob(JobExecution execution) {
        // 向Job的Context中设置参数
        ExecutionContext context = execution.getExecutionContext();
        for (Map.Entry<String, Object> entry: params.entrySet()) {
            context.put(entry.getKey(), entry.getValue());
        }
    }

    @Override
    public void afterJob(JobExecution execution) {
        
    }

    public static class Builder {
        private final JobGlobalVariable inner;
        private final Map<String, Object> params;
        private Builder() {
            params = new HashMap<>();
            this.inner = new JobGlobalVariable();
        }

        public static Builder builder() {
            return new Builder();
        }

        public Builder put(String key, Object value) {
            this.params.put(key, value);
            return this;
        }

        public JobGlobalVariable build() {
            this.inner.setParams(this.params);
            return this.inner;
        }
    }
}

/**
 * @description:
 * @author: Mr.Lee
 * @date: 2021-08-23 15:16
 */
public abstract class IncompetentItemReader<T> extends Item implements ItemReader<T> {

    private List<T> source = null;


    /**
     * 从数据源, 读取数据 (具体的业务实现)
     * @return 不可以返回 null, 无数据时, 返回 []
     */
    public abstract List<T> readSource();

    @Override
    public T read() throws Exception, UnexpectedInputException, ParseException, NonTransientResourceException {
        if (Objects.isNull(source) || source.isEmpty()) {
            source = readSource();
        }
        if (source.isEmpty()) {
            return null;
        }
        return source.remove(0);
    }
}

import lombok.extern.slf4j.Slf4j;
import org.springframework.batch.core.configuration.annotation.StepScope;
import org.springframework.batch.item.ExecutionContext;
import org.springframework.stereotype.Component;

import java.util.List;

/**
 * @description:
 * @author: Mr.Lee
 * @date: 2021-08-20 14:32
 */
@Slf4j
public class ContextWriter extends IncompetentItemWriter<String> {

    private String variableName;

    public void setVariableName(String variableName) {
        this.variableName = variableName;
    }

    @Override
    public void write(List<? extends String> items) throws Exception {
        // 将结果放入Context中, 以供NextStep使用
        ExecutionContext context = stepExecution.getJobExecution().getExecutionContext();
        context.put(variableName, items.get(0));
        log.info("Context Writer: " + items);
    }
}

import com.google.common.collect.Lists;
import org.springframework.batch.core.configuration.annotation.StepScope;
import org.springframework.batch.item.ExecutionContext;
import org.springframework.stereotype.Component;

import java.util.Collections;
import java.util.List;

/**
 * @description:
 * @author: Mr.Lee
 * @date: 2021-08-20 11:47
 */
public class ContextReader extends IncompetentItemReader<String> {

    private boolean isRunning = true;

    private String variableName;

    public void setVariableName(String variableName) {
        this.variableName = variableName;
    }

    @Override
    public List<String> readSource() {
        if (isRunning) {
            isRunning = false;
            // 动态配置: 从 Job 的 Context中拿到参数
            ExecutionContext context = stepExecution.getJobExecution().getExecutionContext();
            return Lists.newArrayList(context.getString(variableName));
        }
        return Collections.emptyList();
    }
}

创建Step

@StepScope
public Step generateStep(JSONObject buildStep) throws Exception {
    SimpleStepBuilder<Object, Object> builder = steps.get(buildStep.getString("name")).chunk(buildStep.getInteger("chunk"));
    builder.reader(itemsConfiguration.generateReader(buildStep.getJSONObject("reader")));
    builder.processor(itemsConfiguration.generateProcessor(buildStep.getJSONObject("processor")));
    builder.writer(itemsConfiguration.generateWriter(buildStep.getJSONObject("writer")));
    // 添加监听器, 主要是为了调用Item里的@BeforeStep方法
    builder.listener(new CompositeStepExecutionListener());
    return builder.build();
}

private JobExecutionDecider decider() {

public TaskExecutor taskExecutor(String poolNmae){
    return new SimpleAsyncTaskExecutor(poolNmae);
}

private Flow flow3() {
    return new FlowBuilder<Flow>("flow3")
        .start(stepFactory.get("step4").tasklet(new TaskletStep(4)).build())
        .build();
}

public Flow parallelFlow() {
    return new FlowBuilder<SimpleFlow>("parallelFlow")
        .split(taskExecutor())
        .add(flow1(), flow2())
        .build();
}
