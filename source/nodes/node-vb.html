Public Sub Voluation()
Randomize Time
Dim i As Integer, j As Integer
 For i = 1 To 54
  If i >= 53 Then
   Pai(i).mz = 14
  Else
   If i Mod 13 = 0 Then
    Pai(i).mz = 13
   Else
    Pai(i).mz = i Mod 13
   End If
  End If

  Select Case Pai(i).mz
   Case Is = 7
    Pai(i).mz = 25
   Case Is = 14
    Pai(i).mz = 24
   Case Is = 5
    Pai(i).mz = 23
   Case Is = 2
    Pai(i).mz = 22
   Case Is = 3
    Pai(i).mz = 21
  End Select
   
  Select Case Pai(i).mz

    Pai(i).fz = 5
   Case Is = 10
    Pai(i).fz = 10
   Case Is = 13
    Pai(i).fz = 10

 Next i

  x(i) = Int(Rnd * 54) + 1
  For j = 1 To i - 1
   If x(i) = x(j) Then
    i = i - 1
    Exit For
   End If
  Next j
 Next i
