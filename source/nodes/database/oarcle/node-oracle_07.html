  FROM (SELECT ('1A000000002702758284, 1A000000002702758285, 1A000000002702758334') BHALL
          FROM DUAL)
CONNECT BY ROWNUM <= LENGTH(BHALL) - LENGTH(REPLACE(BHALL, ',', '')) + 1
