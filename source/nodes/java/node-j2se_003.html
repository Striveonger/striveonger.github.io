Base64 转 图片

少废话, 直接看代码

public static boolean base64ToImage(String base64, String url) {
	if (base64 == null || "".equals(base64)) { // 图像数据为空
		return false;
	}
	BASE64Decoder decoder = new BASE64Decoder();
	try {
		// Base64解码
		byte[] b = decoder.decodeBuffer(base64);
		for (int i = 0; i < b.length; ++i) {
			if (b[i] < 0) {// 调整异常数据
				b[i] += 256;
			}
		}
		// 生成jpeg图片
		String imgFilePath = url;// 新生成的图片
		OutputStream out = new FileOutputStream(imgFilePath);
		out.write(b);
		out.flush();
		out.close();
		return true;
	} catch (Exception e) {
		e.printStackTrace();
		return false;
	}
}
