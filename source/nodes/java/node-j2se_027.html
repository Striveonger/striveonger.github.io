# 二. 扫描项目

# 1. 单元测试覆盖率

参考官方文档中的做法, 由于项目是多模块,

在父级pom.xml 中引入 jacoco-maven-plugin 插件

<plugin>
    <groupId>org.jacoco</groupId>
    <artifactId>jacoco-maven-plugin</artifactId>
    <version>0.8.7</version>
    <executions>
        <execution>
            <id>prepare-agent</id>
            <goals>
                <goal>prepare-agent</goal>
            </goals>
        </execution>
        <execution>
            <id>report</id>
            <goals>
                <goal>report</goal>
            </goals>
            <configuration>
                <formats>
                    <format>XML</format>
                </formats>
            </configuration>
        </execution>
    </executions>
</plugin>

新建 omm-jacoco-report 子模块

<project xmlns="http://maven.apache.org/POM/4.0.0" xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
         xsi:schemaLocation="http://maven.apache.org/POM/4.0.0 http://maven.apache.org/xsd/maven-4.0.0.xsd">
    <modelVersion>4.0.0</modelVersion>
    <parent>
        <groupId>com.cecbrain</groupId>
        <artifactId>operate-monitor-middleware</artifactId>
        <version>1.4.0</version>
    </parent>

    <artifactId>omm-jacoco-report</artifactId>


    <description>Aggregate Coverage Report</description>
    <!-- 以下是要跑测试覆盖率的子模块 -->
    <dependencies>
        <dependency>
            <groupId>com.cecbrain</groupId>
            <artifactId>omm-gateway</artifactId>
            <version>${project.version}</version>
        </dependency>
        <dependency>
            <groupId>com.cecbrain</groupId>
            <artifactId>omm-indicator-data</artifactId>
            <version>${project.version}</version>
        </dependency>
        <dependency>
            <groupId>com.cecbrain</groupId>
            <artifactId>omm-monitor-view</artifactId>
            <version>${project.version}</version>
        </dependency>
        <dependency>
            <groupId>com.cecbrain</groupId>
            <artifactId>omm-alert-manage</artifactId>
            <version>${project.version}</version>
        </dependency>
        <dependency>
            <groupId>com.cecbrain</groupId>
            <artifactId>omm-dispose-manage</artifactId>
            <version>${project.version}</version>
        </dependency>
    </dependencies>
    <build>
        <plugins>
            <plugin>
                <groupId>org.jacoco</groupId>
                <artifactId>jacoco-maven-plugin</artifactId>
                <executions>
                    <execution>
                        <id>report-aggregate</id>
                        <phase>verify</phase>
                        <goals>
                            <goal>report-aggregate</goal>
                        </goals>
                    </execution>
                </executions>
                <configuration>
                    <!-- 排除报告中不需要的类 -->
                    <excludes>
                        <!-- omm-gateway -->
                        <exclude>com/cecbrain/omm/gateway/config/*.class</exclude>
                        <exclude>com/cecbrain/omm/gateway/filter/*.class</exclude>
                        <exclude>com/cecbrain/omm/gateway/utils/*.class</exclude>
                        <exclude>com/cecbrain/omm/gateway/web/*.class</exclude>
                        <!-- omm-alert-manage -->

                        <!-- omm-dispose-manage -->
                        <exclude>com/cecbrain/omm/dispose/registry/**/*.class</exclude>

                        <!-- omm-indicator-data -->
                        <exclude>com/cecbrain/omm/indicator/constant/*.class</exclude>
                        <exclude>com/cecbrain/omm/indicator/web/dto/*.class</exclude>
                        <exclude>com/cecbrain/omm/indicator/web/utils/*.class</exclude>
                        <exclude>com/cecbrain/omm/indicator/web/schedule/*.class</exclude>
                        <exclude>com/cecbrain/omm/indicator/remote/**/*.class</exclude>
                        <!-- omm-monitor-view -->
                    </excludes>
                    <!-- 白名单 -->
                    <includes>
                        <include>com/cecbrain/omm/**/*.class</include>
                    </includes>
                </configuration>
            </plugin>
        </plugins>
    </build>
</project>
