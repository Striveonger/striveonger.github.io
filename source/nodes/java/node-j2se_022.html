接下来, 看下自定义ClassLoader的 loadClass() 方法

@Override
protected synchronized Class<?> loadClass(String name, boolean resolve) throws ClassNotFoundException, ClassFormatError {
    // JDK标准库应由父类加载
    if (isJDKStandardClassName(name))
      	// 直接用引导类加载器加载并返回
        return Class.forName(name, resolve, getParentClassLoader());

    // -拷贝自java.net.URLClassLoader.FactoryURLClassLoader
    SecurityManager sm = System.getSecurityManager();
    if (sm != null) {
        int i = name.lastIndexOf('.');
        if (i != -1) {
            sm.checkPackageAccess(name.substring(0, i));
        }
    }
    // -

    Class<?> clazz = findLoadedClass(name);

    if (clazz == null) {
        try {
            clazz = findClass(name);
        } catch (ClassNotFoundException e) {
        } catch (ClassFormatError e) {
            // 类加载出错（比如版本不兼容），则抛出
            throw e;
        }
    }

    // 强制在此加载的类
    if (clazz == null && this.outsideForceLoads != null && this.outsideForceLoads.contains(name)) {
      	// getParentClassLoader() 获取到的是应用类加载器(该方法代码就不粘贴了)
      	// 使用 getResourceAsStream() 原因是, 只用应用类加载器加载一个class文件, 避免产生递归加载的风险
        InputStream in = getParentClassLoader().getResourceAsStream(classNameToPath(name));

        if (in == null)
            throw new ClassNotFoundException(name);

        byte[] bytes = null;

        try {
            bytes = IOUtil.getBytes(in);
        } catch (IOException e) {
            throw new ClassNotFoundException(name, e);
        } finally {
            IOUtil.close(in);
        }

        clazz = defineClass(name, bytes, 0, bytes.length);
    }

    // 此时仅代理父类加载JDK扩展库，其他都应由此类加载器加载(这样, 就只绕过了应用类加载器)
    if (clazz == null && isJDKExtClassName(name)) {
        ClassLoader parent = getParentClassLoader();
        clazz = Class.forName(name, false, parent);
    }

    if (clazz == null)
        throw new ClassNotFoundException(name);

    if (resolve)
        resolveClass(clazz);
    
    return clazz;
}

/**
 * 初始化 pathDriverClassLoader 和 driverTool
 *
 * @throws PathDriverFactoryException
 */
public synchronized void init() throws PathDriverFactoryException {
    if (this.pathDriverClassLoader != null)
        return;

    // 初始化ClassLoader
    this.pathDriverClassLoader = new PathDriverClassLoader(this.path);

    try {
      	// 初始化 DriverTools.
        Class<?> driverToolClass = this.pathDriverClassLoader.loadClass(DriverTool.class.getName());
        this.driverTool = driverToolClass.newInstance();
    } catch (ClassNotFoundException | InstantiationException | IllegalAccessException e) {
        throw new PathDriverFactoryException(e);
    }
}

public synchronized Driver getDriver(String driverClassName) throws PathDriverFactoryException {
  	try {
        // 使用 pathDriverClassLoader 去加载指定的驱动类
        Class.forName(driverClassName, true, this.pathDriverClassLoader);
    } catch (ClassNotFoundException e) {
        throw new DriverNotFoundException(this.path.getPath(), driverClassName, e);
    } catch (ClassFormatError e) {
        throw new DriverClassFormatErrorException(e);
    } catch (Throwable t) {
        throw new PathDriverFactoryException(t);
    }

    try {
        Driver driver = (Driver) this.driverTool.getClass().getMethod("getDriver", String.class).invoke(this.driverTool, driverClassName);
        if (driver == null) {
            throw new PathDriverFactoryException("No Driver named [" + driverClassName + "] found in [" + this.path + "]");
        }
        return driver;
    } catch (IllegalArgumentException | SecurityException | IllegalAccessException | InvocationTargetException | NoSuchMethodException e) {
        throw new PathDriverFactoryException(e);
    }
}

public Driver getDriver(String driverClassName) {
    Set<Driver> drivers = getDrivers();
    for (Driver driver : drivers) {
      	// 判断类是否是指定类名的类或者子类
        if (isAssignableClass(driver.getClass(), driverClassName))
            return driver;
    }
    return null;
}

public Set<Driver> getDrivers() {
    Set<Driver> myDrivers = new HashSet<Driver>();
    // 这里是重点哦
    Enumeration<Driver> drivers = DriverManager.getDrivers();
    while (drivers.hasMoreElements()) {
        Driver driver = drivers.nextElement();
        myDrivers.add(driver);

    return myDrivers;
}

看一下, 这个所谓的重点 DriverManager.getDrivers() 吧:

@CallerSensitive
public static Enumeration<Driver> getDrivers() {
		// 通过SPI加载应用类加载器中的JDBC驱动(这里不是重点, 咱们自己的应用类加载器中没有JDBC的驱动)
  	ensureDriversInitialized();
  	return Collections.enumeration(getDrivers(Reflection.getCallerClass()));
}

private static List<Driver> getDrivers(Class<?> callerClass) {
    List<Driver> result = new ArrayList<>();
    // Walk through the loaded registeredDrivers.
    for (DriverInfo aDriver : registeredDrivers) {
      	// If the caller does not have permission to load the driver then skip it.
      	// 如果调用者没有加载驱动程序的权限, 则跳过它.
      	// 这里需要注意 callerClass, 调用者的Calss(DriverTool.class)
        if (isDriverAllowed(aDriver.driver, callerClass)) {
            result.add(aDriver.driver);
        }
    }
    return result;
}

private static boolean isDriverAllowed(Driver driver, Class<?> caller) {
  	// 显然 caller != null; 且 caller.getClassLoader() 是 PathDriverFactory 中所维护的 pathDriverClassLoader
    ClassLoader callerCL = caller != null ? caller.getClassLoader() : null;
    return isDriverAllowed(driver, callerCL);
}

private static boolean isDriverAllowed(Driver driver, ClassLoader classLoader) {
    boolean result = false;
    if (driver != null) {
        Class<?> aClass = null;
        try {
          	// 加载自己的ClassLoder中的, 指定Driver类
            aClass =  Class.forName(driver.getClass().getName(), true, classLoader);
        } catch (Exception ex) {
            result = false;
        }
      	// 判断是不是一个Class对象
        result = ( aClass == driver.getClass() ) ? true : false;
    }
    return result;
}
