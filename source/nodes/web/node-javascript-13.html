/**
 *  判断复选框是否为空
 *    需要传入 checkBox的name属性
 *    返回true,为空
 *    返回false, 不为空
 */
function checkBoxOrRadioIsNull(name) {

    var arr = document.getElementsByName(name);
    var i;
    for (i = 0; i < arr.length; i++) {
        if(arr[i].checked) {
            break;
        }
    }
    if(i < arr.length) {
        return false;
    } else {
        return true;
    }
}

/**
 *  选择下拉框的公用方法
 *      需要传入 select的id属性 和 需要选中的值
 *
 */

function selectPicking(id, val) {
    var arr = document.getElementById(id).options;
    for (var i = 0; i < arr.length; i++) {
        if(arr[i].value == val) {
            document.getElementById(id).selectedIndex = i;
            break;
        }
    }
}
