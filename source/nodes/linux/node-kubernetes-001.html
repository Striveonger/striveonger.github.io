启动类

package com.striveonger.demo;

import java.io.File;
import java.io.FileInputStream;
import java.io.InputStream;
import java.nio.file.Files;
import java.nio.file.Paths;
import java.util.HashMap;
import java.util.Map;

import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

import io.kubernetes.client.openapi.ApiClient;
import io.kubernetes.client.openapi.Configuration;
import io.kubernetes.client.openapi.apis.CoreV1Api;
import io.kubernetes.client.openapi.models.V1Node;
import io.kubernetes.client.openapi.models.V1NodeList;
import io.kubernetes.client.openapi.models.V1Pod;
import io.kubernetes.client.openapi.models.V1PodList;
import io.kubernetes.client.util.Config;


@RestController
@SpringBootApplication
public class K3sApplication {

	public static void main(String[] args) {
		SpringApplication.run(K3sApplication.class, args);
	}

	@GetMapping("hello")
	public Map<String, String> hello() {
		// https://github.com/kubernetes-client/java/wiki/3.-Code-Examples
		try {
			// 读取服务账户的访问令牌
            String token = new String(Files.readAllBytes(Paths.get("/var/run/secrets/kubernetes.io/serviceaccount/token")));            
            // 读取用于验证 Kubernetes API 服务器的根证书
            InputStream caCert = new FileInputStream(new File("/var/run/secrets/kubernetes.io/serviceaccount/ca.crt"));
            // 设置 Kubernetes 客户端的访问令牌和根证书
            ApiClient client = Config.defaultClient();
            client.setAccessToken(token);
			client.setSslCaCert(caCert);
			
			Configuration.setDefaultApiClient(client);

			CoreV1Api api = new CoreV1Api();
			Map<String, String> map = new HashMap<>();
			
			// 命令: kubectl get nodes --all-namespaces -o wide
			V1NodeList nodes = api.listNode(null, null, null, null, null, null, null, null, null, null);
			for (V1Node node : nodes.getItems()) {
				String nodeName = node.getMetadata().getName();
				String nodeIP = node.getMetadata().getAnnotations().get("flannel.alpha.coreos.com/public-ip");
				System.out.println("Node Name: " + nodeName + ", Node IP: " + nodeIP);
				map.put(nodeName, nodeIP);
			}
			Map<String, String> result = new HashMap<>();
			// 命令: kubectl get pods --all-namespaces -o wide
			V1PodList list = api.listPodForAllNamespaces(null, null, null, null, null, null, null, null, null, false);
			for (V1Pod item : list.getItems()) {
				String nodeName = item.getSpec().getNodeName();
				String podName = item.getMetadata().getName();
				String podIp = item.getStatus().getPodIP();
				System.out.println("Node Name: " + nodeName
							+ ", Pod Name: " + podName
							+ ", Pod IP: " + podIp);
				result.put(podIp, map.getOrDefault(nodeName, "unknown"));
			}
			return result;
		} catch(Exception e) {
			e.printStackTrace();
			return Map.of();
		}
	}
}

# 4. Kubernetes 认证, 授权, 准入控制

  1. 认证（Authentication）：API Server 可以支持多种身份认证方式，如证书、静态令牌、动态令牌（Token）和基本认证等。使用 TLS 证书认证是最安全的身份认证方式。
  2. 授权（Authorization）：Kubernetes 集群中所有的资源都有自己的 API 对象，如 Pod、Service、ReplicationController 等，而这些对象都有读写操作权限。因此，API Server 还需要检查请求的用户是否有足够的授权，才会处理请求。
  3. 准入控制（Admission Control）：准入控制机制是 Kubernetes API Server 中一个非常强大的功能，它允许对进入 Kubernetes 集群的请求做出控制。API Server 会在对象（如 Pod、Node、Service 等）被保存到 etcd 数据库之前，对该对象进行检查和修改。准入控制可用于实施强制性的数据验证、安全功能、自动化操作等功能。
  4. 访问控制列表（Access Control List, ACL）：ACL 可以通过配置文件中的用户和角色绑定来限制用户的访问权限。通过定义 Role 和 ClusterRole，以及 RoleBinding 和 ClusterRoleBinding 对象，可以实现对 Kubernetes 集群中不同用户的访问控制。

# 5. Kubernetes RBAC

基于角色（Role）的访问控制（RBAC）是一种基于组织中用户的角色来调节控制对计算机或网络资源的访问的方法。

RBAC 鉴权机制使用 rbac.authorization.k8s.io API 组来驱动鉴权决定， 允许你通过 Kubernetes API 动态配置策略。

要启用 RBAC，在启动 API 服务器时将 --authorization-mode 参数设置为一个逗号分隔的列表并确保其中包含 RBAC。

在 Kubernetes 中，Role 和 ClusterRole 都是用于定义一组权限的策略对象，但它们在作用范围和可授予的权限类型方面存在一些关键差异。

作用范围

  • Role: Role 的作用范围限制在一个特定的命名空间内。这意味着 Role 授予的权限只能用于访问该命名空间中的资源。
  • ClusterRole: ClusterRole 的作用范围涵盖整个集群。这意味着 ClusterRole 授予的权限可以用于访问任何命名空间中的资源，以及一些只有集群管理员才能访问的集群范围资源，例如节点和端点。

可授予的权限类型

  • Role:

    Role 可以授予对以下资源类型的权限：

    • 命名空间内Pod、Service、Deployment、ReplicaSet、ConfigMap、Secret 等资源
    • 非资源类型，例如 /healthz

  • ClusterRole:

    ClusterRole 除了可以授予 Role 可以授予的所有权限之外，还可以授予对以下资源类型的权限：

    • 集群范围资源，例如节点
    • 非资源类型，例如 /apis/rbac.authorization.k8s.io/v1/clusterroles

其他区别

  • 创建和更新限制: 创建或更新 Role 的权限仅限于具有该命名空间中 rbac.authorization.k8s.io/create 或 rbac.authorization.k8s.io/update 权限的用户。创建或更新 ClusterRole 的权限仅限于具有集群范围 rbac.authorization.k8s.io/escalate 权限的用户。
  • 优先级: 如果在同一主体上绑定了 Role 和 ClusterRole，则 ClusterRole 授予的权限将优先于 Role 授予的权限。

总结

总而言之，Role 和 ClusterRole 都是用于定义 Kubernetes 中用户或服务帐户权限的策略对象。但是，它们在作用范围和可授予的权限类型方面存在一些关键差异。

  • 使用 Role 来授予对特定命名空间中资源的访问权限。
  • 使用 ClusterRole 来授予对整个集群中资源的访问权限，包括集群范围资源和非资源类型。

以下是一些有关何时使用 Role 和 ClusterRole 的建议：

  • 如果您需要授予对特定命名空间中资源的访问权限，请使用 Role。
  • 如果您需要授予对整个集群中资源的访问权限，请使用 ClusterRole。
  • 如果您需要授予对集群范围资源或非资源类型的访问权限，请使用 ClusterRole。

一些最常见的 API 组：

  • apps: 此 API 组包含用于管理应用程序的资源，例如 deploy 和 statefulset。
  • core: 此 API 组包含核心资源，例如节点、Pod 和服务。
  • extensions: 此 API 组包含已弃用的资源，例如 Ingress 和部署。
  • networking.k8s.io: 此 API 组包含用于管理网络的资源，例如网络策略和 IngressClass。
  • rbac.authorization.k8s.io: 此 API 组包含用于管理基于角色的访问控制 (RBAC) 的资源。
  • storage.k8s.io: 此 API 组包含用于管理存储的资源，例如永久卷和永久卷声明。

            String namespace = pod.getMetadata().getNamespace();

            String namespace = deployment.getMetadata().getNamespace();
            String name = deployment.getMetadata().getName();

            map.put(name, replicas);
        }
        log.info("map: {}", JSON.toJSONString(map));
    }
