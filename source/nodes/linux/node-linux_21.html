# 一. 搭建基础环境

# 1. 创建支持SSH的镜像

做为被控制节点来使用...

# 使用基础镜像
FROM ubuntu

# 设置时区
RUN ln -snf /usr/share/zoneinfo/Asia/Shanghai /etc/localtime && echo 'Asia/Shanghai' > /etc/timezone

# 安装SSH服务器
RUN apt-get update && apt-get install -y openssh-server sshpass
RUN mkdir /var/run/sshd

# 创建一个新的用户
RUN useradd -m strivonger

# 设置密码
RUN echo "strivonger:123456" | chpasswd
RUN echo "root:root" | chpasswd

# 允许strivonger用户使用sudo命令
RUN echo "strivonger ALL=(ALL) NOPASSWD:ALL" >> /etc/sudoers

# 允许SSH root 登录
RUN sed -i 's/#PermitRootLogin prohibit-password/PermitRootLogin yes/' /etc/ssh/sshd_config

# 启动SSH服务
EXPOSE 22
CMD ["/usr/sbin/sshd", "-D"]

# 二. 安装并配置Ansible

# 1. 安装Ansible

进入容器: docker exec -ti ansible /bin/bash

最小化安装: pip install ansible-core

安装版本 ansible-core-2.11.12

安装 sshpass: apt install sshpass -y

SSH 登陆不能在命令行中指定密码, sshpass 的出现则解决了这一问题. 它允许你用 -p 参数指定明文密码, 然后直接登录远程服务器, 它支持密码从命令行, 文件, 环境变量中读取

# 2. 执行 ansible-playbook

---
- hosts:
    - "10.13.144.204"
    - "10.13.144.205"
    - "10.13.144.206"
  name: "DynamicPlaybook"
  gather_facts: false
  vars:
    x: 1
    o:
      "y": 2
      z: 3

  tasks:
    - name: out_datetime_to_file
      shell: 'echo "$(date +%F) $(date +%T) ......x: {{ x }}, y: {{ o.y }}, z: {{ o.z }}" >> ~/hello.txt'

    - name: copy_file
      copy:
        src: '~/test.txt'
        dest: '~/test.txt'
        mode: 'u=rw,g=rw,o=rw'

    - name: cat_file
      shell: 'cat ~/test.txt >> ~/hello.txt'
