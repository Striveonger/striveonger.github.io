启动类

package com.striveonger.demo;

import java.io.File;
import java.io.FileInputStream;
import java.io.InputStream;
import java.nio.file.Files;
import java.nio.file.Paths;
import java.util.HashMap;
import java.util.Map;

import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

import io.kubernetes.client.openapi.ApiClient;
import io.kubernetes.client.openapi.Configuration;
import io.kubernetes.client.openapi.apis.CoreV1Api;
import io.kubernetes.client.openapi.models.V1Node;
import io.kubernetes.client.openapi.models.V1NodeList;
import io.kubernetes.client.openapi.models.V1Pod;
import io.kubernetes.client.openapi.models.V1PodList;
import io.kubernetes.client.util.Config;


@RestController
@SpringBootApplication
public class K3sApplication {

	public static void main(String[] args) {
		SpringApplication.run(K3sApplication.class, args);
	}

	@GetMapping("hello")
	public Map<String, String> hello() {
		// https://github.com/kubernetes-client/java/wiki/3.-Code-Examples
		try {
			// 读取服务账户的访问令牌
            String token = new String(Files.readAllBytes(Paths.get("/var/run/secrets/kubernetes.io/serviceaccount/token")));
            
            // 读取用于验证 Kubernetes API 服务器的根证书
            InputStream caCert = new FileInputStream(new File("/var/run/secrets/kubernetes.io/serviceaccount/ca.crt"));
            // 设置 Kubernetes 客户端的访问令牌和根证书
            ApiClient client = Config.defaultClient();
            client.setAccessToken(token);
			client.setSslCaCert(caCert);
			
			Configuration.setDefaultApiClient(client);

			CoreV1Api api = new CoreV1Api();
			Map<String, String> map = new HashMap<>();
			
			// 命令: kubectl get nodes --all-namespaces -o wide
			V1NodeList nodes = api.listNode(null, null, null, null, null, null, null, null, null, null);
			for (V1Node node : nodes.getItems()) {
				String nodeName = node.getMetadata().getName();
				String nodeIP = node.getMetadata().getAnnotations().get("flannel.alpha.coreos.com/public-ip");
				System.out.println("Node Name: " + nodeName + ", Node IP: " + nodeIP);
				map.put(nodeName, nodeIP);
			}
			Map<String, String> result = new HashMap<>();
			// 命令: kubectl get pods --all-namespaces -o wide
			V1PodList list = api.listPodForAllNamespaces(null, null, null, null, null, null, null, null, null, false);
			for (V1Pod item : list.getItems()) {
				String nodeName = item.getSpec().getNodeName();
				String podName = item.getMetadata().getName();
				String podIp = item.getStatus().getPodIP();
				System.out.println("Node Name: " + nodeName
							+ ", Pod Name: " + podName
							+ ", Pod IP: " + podIp);
				result.put(podIp, map.getOrDefault(nodeName, "unknown"));
			}
			return result;
		} catch(Exception e) {
			e.printStackTrace();
			return Map.of();
		}
	}
}

添加资源权限

---
apiVersion: rbac.authorization.k8s.io/v1
kind: Role
metadata:
  namespace: default
  name: admin-role
rules:
- apiGroups: [""]
  # resources: ["pods", "nodes"]
  # verbs: ["list"]
  resources: ["*"]
  verbs: ["*"]

---
apiVersion: rbac.authorization.k8s.io/v1
kind: RoleBinding
metadata:
  namespace: default
  name: admin-role-binding
subjects:
- kind: ServiceAccount
  name: system
  namespace: default # ServiceAccount是某个服务/资源/程序使用的, ServiceAccount作为K8S内部的某种资源，是存在于某个命名空间之中的，在不同命名空间中的同名ServiceAccount被认为是不同的资源
roleRef:
  kind: Role
  name: admin-role
  apiGroup: ""
