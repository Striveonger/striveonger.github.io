# 完整的描述文件

@startuml
'https://plantuml.com/class-diagram

'top to bottom direction
left to right direction

' 水
class Water {
    int H2O
}

' 空气
class Air {
    int O2
}

' 动物
abstract Animal <<abstract>> {
    ' 是否有生命
    boolean isBeing

    ' 代谢
    void metabolism(Water water, Air air)

    ' 繁育
    abstract Animal breed()
}

' 鸟
abstract Fowl <<abstract>> {

    Feather feather

    ' 繁育
    Animal breed()
    ' 下蛋
    abstract Animal layEggs()

}

' 羽毛
class Feather

' 翅膀
class Wings

' 飞
interface Fly <<interface>> {
    void fly()
}

' 大雁
class WildGoose {
    Animal layEggs()
    void fly()
}

' 鸭子
class Duck {
    Animal layEggs()
}

' 企鹅
class Penguin {
    Animal layEggs()
    void fly()
}

' 组合与聚合的描述
note "翅膀可以多一根或少一根羽毛,\n但是鸟多一个翅膀或少一个翅膀你试试..." as Note1



' 开始描述类之间的关系

' 依赖关系(鸟依赖水和空气)
Animal ..> Air: 依赖关系(鸟依赖空气)
Animal ..> Water: 依赖关系(鸟依赖水)

' 继承关系(鸟继承动物)
Fowl --|> Animal: 继承关系(鸟继承动物)

' 组合关系(鸟需要一对翅膀)
Fowl "1" *--> "2" Wings: 组合关系(鸟需要一对翅膀, 是一种强"拥有"的关系)

' 聚合关系(翅膀需要羽毛)
Wings "1" o--> "n" Feather: 聚合关系(翅膀包含多根羽毛, 是一种弱"拥有"的关系)

' 添加描述
Wings .. Note1
Note1 .. Feather

' 继承
WildGoose --|> Fowl
Duck --|> Fowl
Penguin --|> Fowl

' 实现接口
WildGoose ..|> Fly
Penguin ..|> Fly
