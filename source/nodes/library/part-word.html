import com.google.common.collect.Lists;
import lombok.extern.slf4j.Slf4j;
import org.lionsoul.jcseg.extractor.impl.TextRankKeywordsExtractor;
import org.lionsoul.jcseg.tokenizer.core.*;

import java.io.IOException;
import java.io.StringReader;
import java.util.List;
import java.util.Objects;
import java.util.stream.Stream;

import static java.util.stream.Collectors.joining;
import static java.util.stream.Collectors.toList;

/**
 * @description: 分词工具
 * @author: Mr.Lee
 * @date: 2019-08-02 10:48
 */
@Slf4j
public class PartWordUtil {

	private ADictionary dic;

	private JcsegTaskConfig config;

	private static PartWordUtil instance = null;

	/**
	 * 构建分词工具类
	 * @return 分词工具类
	 */
	public static PartWordUtil getInstance() {
		if (instance == null) {
			instance = new PartWordUtil(null);
		}
		return instance;
	}

	/**
	 * 构建分词工具类-加载自定义词库
	 * @param customWordsPath 自定义的词库路径
	 * @return 分词工具类
	 */
	public static PartWordUtil getInstance(String[] customWordsPath) {
		if (instance == null) {
			instance = new PartWordUtil(customWordsPath);
		}
		return instance;
	}

	private PartWordUtil(String[] customWordsPath) {
		// 创建JcsegTaskConfig分词配置实例，自动查找加载jcseg.properties配置项来初始化
		config = new JcsegTaskConfig(true);

        // 设置过滤停止词
		config.setClearStopwords(true);
		// 关闭同义词追加
		config.setAppendCJKSyn(false);
		// 去除不识别的词条
		config.setKeepUnregWords(false);

        // 会忽略系统词库(可能在特殊场景下会用到)
		// path是文件夹路径, 且词库文件以`lex-`开头, 以`.lex`结尾
		// config.setLexiconPath(customWordsPath);

		// 创建默认单例词库实现，并且按照config配置加载词库
		dic = DictionaryFactory.createSingletonDictionary(config);

		// 设置自定义词库
		if (Objects.nonNull(customWordsPath)) {
			Stream.of(customWordsPath).forEach(path -> {
				try {
					dic.load(path);
				} catch (IOException e) {
					log.error("自定义词库路径无效, 忽略此词库: {}", path);
				}
			});
		}
	}

	/**
	 * 对数据进行分词
	 * @param data 需要分词的数据
	 * @param maxNum 分词的最大数量
	 * @return
	 */
	public synchronized String[] part(String[] data, int maxNum) {

		List<String> words = Lists.newArrayList();
		String content = Stream.of(data).filter(Objects::nonNull).collect(joining(","));
		try {
			// 依据给定的ADictionary和JcsegTaskConfig来创建ISegment
			ISegment segment = SegmentFactory.createJcseg(JcsegTaskConfig.DETECT_MODE, new Object[]{config, dic});

			// 构建取词器
			TextRankKeywordsExtractor extractor = new TextRankKeywordsExtractor(segment);

			// 设置PageRank算法, 最大迭代次数.
			// extractor.setMaxIterateNum(100);

			// 设置TextRank计算窗口大小.
			// extractor.setWindowSize(5);

			// 设置最大返回的关键词个数
			extractor.setKeywordsNum(maxNum);

			words = extractor.getKeywords(new StringReader(content));
		} catch (JcsegException e) {
			log.error("生成取词器失败...");
			e.printStackTrace();
		} catch (IOException e) {
			log.error("取词失败...");
			e.printStackTrace();
		}
		return words.stream().distinct().collect(toList()).toArray(new String[]{});
	}
}
