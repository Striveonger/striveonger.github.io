# Demo入口

package xyz.striveonger.demo.concurrent.chapter_03.juc.lock;

import java.util.concurrent.locks.Condition;
import java.util.concurrent.locks.Lock;
import java.util.concurrent.locks.LockSupport;
import java.util.concurrent.locks.ReentrantLock;

/**
 * @description: 线程A: A~Z, 线程B: 1~26, 输出: A1B2...Z26
 * @author: Mr.Lee
 * @date: 2021-06-01 03:21
 */
public class ConditionExample {

    public static void main(String[] args) {
        final Lock lock = new ReentrantLock();
        // 执行这货的方法前, 必须先持有一把锁, 不管是Lock还是synchronized 都行...是锁就行
        // 还有还有....tm不要把 await 和 wait 用混了....别问我是怎么知道的....
        final Condition condition = lock.newCondition();
        Thread t2 = new Thread(() -> {
            LockSupport.park(); // 为确保 t1先持有锁
            lock.lock();
            try {
                for (int i = 0; i < 26; i++) {
                    System.out.println(i + 1);
                    condition.signal(); // 唤醒挂起的t1
                    if (i < 25) {
                        condition.await(); // 挂起t2 & 释放lock(最后一次, 不再挂起t2)
                    }
                }
            } catch (InterruptedException e) {
                e.printStackTrace();
            } finally {
                lock.unlock();
            }
        }, "t2");

        Thread t1 = new Thread(() -> {
            lock.lock();
            LockSupport.unpark(t2);
            try {
                for (int i = 0; i < 26; i++) {
                    System.out.print((char) (i + 65));
                    condition.signal(); // 唤醒挂起的t2
                    condition.await(); // 挂起t1 & 释放lock
                }
            } catch (InterruptedException e) {
                e.printStackTrace();
            } finally {
                lock.unlock();
            }
        }, "t1");

        t1.start();
        t2.start();
    }
}

# condition.await()

挂起当前线程

public final void await() throws InterruptedException {
    if (Thread.interrupted()) {
        // 如果线程被打断过, 直接抛异常走了...
        throw new InterruptedException();
    }
    // 当前condition队列中的尾部
    Node node = addConditionWaiter();
    // 获取当前锁重入次数, 并释放持有锁
    int savedState = fullyRelease(node);
    int interruptMode = 0;
    while (!isOnSyncQueue(node)) { // 判断是当前节点, 是否在锁的同步队列中
        // 直接阻塞掉, 当前线程
        LockSupport.park(this); // 街道信号, 
        // 被唤醒后, 第一件事, 判断阻塞过程中, 是否被打断过.
        if ((interruptMode = checkInterruptWhileWaiting(node)) != 0) {
            break;
        }
    }
    // 尝试争抢锁: 争抢到锁 且 没有被打断
    if (acquireQueued(node, savedState) && interruptMode != THROW_IE) {
        // 线程继续运行
        interruptMode = REINTERRUPT;
    }
    if (node.nextWaiter != null) { // clean up if cancelled
        //  踢出当前condition队列中, 状态不等于 CONDITION 的节点
        unlinkCancelledWaiters();
    }
   	
    if (interruptMode != 0) {
        reportInterruptAfterWait(interruptMode);
    }
}

// 当前condition队列中的尾部, 添加node
private Node addConditionWaiter() {
    // 判断当前线程,是不是锁的持有者
    if (!isHeldExclusively()) {
        throw new IllegalMonitorStateException();
    }
    // 拿到当前condition队列中的尾节点
    Node t = lastWaiter;
    // 尾节点状态 != -2时
    if (t != null && t.waitStatus != Node.CONDITION) {
        unlinkCancelledWaiters();
        t = lastWaiter;
    }
    // 新增队列等待节点
    Node node = new Node(Node.CONDITION);
    if (t == null) {
        firstWaiter = node;
    } else {
        t.nextWaiter = node;
    }
    lastWaiter = node;
    return node;
}

// 踢出当前condition队列中, 状态不等于 CONDITION 的节点
private void unlinkCancelledWaiters() {
    // 经典的遍历单行链表的算法了......
    Node t = firstWaiter;
    // 记录最后一个合法的节点.
    Node trail = null;
    while (t != null) {
        Node next = t.nextWaiter;
		
        // 逻辑start...
        // 当前节点的状态 != -2
        if (t.waitStatus != Node.CONDITION) {
            // 踢出t的不合法的后继节点
            t.nextWaiter = null;
            if (trail == null) {
                // 第一个就不合法...
                firstWaiter = next;
            } else {
                // 踢出节点 t
                trail.nextWaiter = next;
            }
            // 把最后一个合法的节点, 给到
            if (next == null) {
                lastWaiter = trail;
            }
        } else {
            // t 合法, 保留在队列中
            trail = t;
        }
		// 逻辑end...
        
        // 依次向后递推
        t = next;
    }
}

int fullyRelease(Node node) {
    try {
        // 获得当前锁的状态
        int savedState = getState();
       	// 释放持有的
        if (release(savedState)) {
            return savedState;
        }
        throw new IllegalMonitorStateException();
    } catch (Throwable t) {
        node.waitStatus = Node.CANCELLED;
        throw t;
    }
}

public final boolean release(int arg) {
    // 尝试释放锁
    if (tryRelease(arg)) { // (state - state == 0) 哈哈~醒醒~~
        // 释放当前线程持有的锁后
        Node h = head;
        if (h != null && h.waitStatus != 0) {
            // unpark 头节点的后继节点
            unparkSuccessor(h);
        }
        return true;
    }
    return false;
}

// 判断node节点, 是否在锁的同步队列中
boolean isOnSyncQueue(Node node) {
    if (node.waitStatus == Node.CONDITION || node.prev == null)
        return false;
    if (node.next != null) // If has successor, it must be on queue
        return true;
    return findNodeFromTail(node);
}

// 从锁的同步队列中, 查找node
boolean findNodeFromTail(Node node) {
    for (Node p = tail;;) {
        if (p == node)
            return true;
        if (p == null)
            return false;
        p = p.prev;
    }
}

// 判断当前线程, 在park过程中, 是否被打断过
private int checkInterruptWhileWaiting(Node node) {
    return Thread.interrupted() ? (transferAfterCancelledWait(node) ? THROW_IE : REINTERRUPT) : 0;
}

// 接到信号, 被唤醒后, 将node节点, 加入到锁的同步队列中
final boolean transferAfterCancelledWait(Node node) {
    // 初始化node节点状态
    if (node.compareAndSetWaitStatus(Node.CONDITION, 0)) {
        // 向锁的同步队列尾部, 添加node节点
        enq(node);
        // 退出等待时抛出InterruptedException
        return true;
    }
    while (!isOnSyncQueue(node))
        Thread.yield();
    // 从等待退出时重新中断
    return false;
}

// 根据不同模式, 执行不同操作
private void reportInterruptAfterWait(int interruptMode) throws InterruptedException {
    if (interruptMode == THROW_IE) {
        // 县城被打断...抛出异常就好了.
        throw new InterruptedException();
    } else if (interruptMode == REINTERRUPT) {
        // 打断当前线程
        selfInterrupt();
    }
}

# condition.signal()

唤醒队列挂起的第一个线程

// ConditionObject 实现Condition接口的方法
public final void signal() {
    if (!isHeldExclusively()) {
        throw new IllegalMonitorStateException();
    }
    Node first = firstWaiter;
    if (first != null) {
        doSignal(first);
    }
}

// 这是 Sync 重写的AQS的方法.
// 用来判断当前调用线程, 是否持有锁
protected final boolean isHeldExclusively() {
    return getExclusiveOwnerThread() == Thread.currentThread();
}
// ConditionObject 的方法
private void doSignal(Node first) {
    do {
        // 判断first的后继节点是否空
        if ((firstWaiter = first.nextWaiter) == null)
            // 为空时, 则first即是第个, 也是最会后一个
            lastWaiter = null;
        first.nextWaiter = null;
    } while (!transferForSignal(first) && (first = firstWaiter) != null);
}

boolean transferForSignal(Node node) {
    // 设置node状态(从Condition状态:-2, 改为初始状态: 0)
    if (!node.compareAndSetWaitStatus(Node.CONDITION, 0)) {
        // 修改失败, 那就重试呗~
        return false;
    }
    // 说白了, 就是从一个队列到另一个队列(下面我画个图, 解释一哈~)
    // 加入到锁的同步队列后, 就不归Condition管理喽~
    Node p = enq(node); // p 是node的前继节点.
    int ws = p.waitStatus; // CLH的结构, 修改前继节点的状态
    // 如果前继节点是取消状态, 或前继节点的状态修改失败.
    if (ws > 0 || !p.compareAndSetWaitStatus(ws, Node.SIGNAL)) {
        // 就关了, 直接就开始执行哦~(有点被震惊到了)
        LockSupport.unpark(node.thread);
    }
    return true;
}

// 向锁的同步队列尾部, 添加node节点
private Node enq(Node node) {
    for (;;) {
        Node oldTail = tail;
        if (oldTail != null) {
            node.setPrevRelaxed(oldTail);
            if (compareAndSetTail(oldTail, node)) {
                oldTail.next = node;
                return oldTail;
            }
        } else {
            // 初始化等待队列
            initializeSyncQueue();
        }
    }
}
