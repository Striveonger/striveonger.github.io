# 3. Node

static final class Node {
    
    // 共享模式结点
    static final Node SHARED = new Node();
    
    // 独占模式结点
    static final Node EXCLUSIVE = null;

    static final int CANCELLED =  1;

    static final int SIGNAL    = -1;

    static final int CONDITION = -2;

    static final int PROPAGATE = -3;

    /**
     * INITAL:      0 - 默认, 新结点会处于这种状态
     * CANCELLED:   1 - 取消, 表示当前节点为取消状态, 可以被踢出队列
     * SIGNAL:     -1 - 发信号, 表示节点被处于阻塞状态;
     * CONDITION:  -2 - Condition专用, 表示当前结点在Condition队列中, 因为等待某个条件而被阻塞了
     * PROPAGATE:  -3 - 传播, 适用于共享模式. (比如连续的读操作结点可以依次进入临界区, 设为PROPAGATE有助于实现这种迭代操作)
     * 
     * waitStatus 表示的是后续结点状态, 这是因为AQS中使用CLH队列实现线程的结构管理, 而CLH结构正是用前一结点某一属性表示当前结点的状态, 这样更容易实现取消和超时功能
     */
    volatile int waitStatus;

    // 前驱节点
    volatile Node prev;

    // 后继节点
    volatile Node next;

    // 节点所包装的线程
    volatile Thread thread;

    // Condition队列使用，存储Condition队列中的后继节点
    Node nextWaiter;

    // 是否为共享模式
    final boolean isShared() {
        return nextWaiter == SHARED;
    }
    
    final Node predecessor() {
        Node p = prev;
        if (p == null)
            throw new NullPointerException();
        else
            return p;
    }

    Node() {}

    Node(Node nextWaiter) {
        this.nextWaiter = nextWaiter;
        THREAD.set(this, Thread.currentThread());
    }

    Node(int waitStatus) {
        WAITSTATUS.set(this, waitStatus);
        THREAD.set(this, Thread.currentThread());
    }

    final boolean compareAndSetWaitStatus(int expect, int update) {
        return WAITSTATUS.compareAndSet(this, expect, update);
    }

    final boolean compareAndSetNext(Node expect, Node update) {
        return NEXT.compareAndSet(this, expect, update);
    }

    final void setPrevRelaxed(Node p) {
        PREV.set(this, p);
    }

    // 可以理解成可以直接操作内存的句柄好了.
    private static final VarHandle NEXT;
    private static final VarHandle PREV;
    private static final VarHandle THREAD;
    private static final VarHandle WAITSTATUS;
    static {
        try {
            // 这基本是就和JDK1.8中的Unsafe是一样的(我用JDK11)
            // 可以直接操作内存的东东
            MethodHandles.Lookup l = MethodHandles.lookup();
            NEXT = l.findVarHandle(Node.class, "next", Node.class);
            PREV = l.findVarHandle(Node.class, "prev", Node.class);
            THREAD = l.findVarHandle(Node.class, "thread", Thread.class);
            WAITSTATUS = l.findVarHandle(Node.class, "waitStatus", int.class);
        } catch (ReflectiveOperationException e) {
            throw new ExceptionInInitializerError(e);
        }
    }
}

# 3. Sync.nonfairTryAcquire()

// 这是Sync中的方法
// 非公平的尝试获得锁(不检查队列, 直接抢夺锁的使用权)
boolean nonfairTryAcquire(int acquires) {
    final Thread current = Thread.currentThread();
    // 获得当前对象的state状态
    int c = getState();
    if (c == 0) {
        // 当前state状态为0时, 尝试把当前对象的state状态改为1
        // 我才不看前面有没有排队的节点呢, 抢到就是我的
        if (compareAndSetState(0, acquires)) {
            // 当前线程, 修改成功后, 
            // 把当前线程设置为持有锁的线程
            setExclusiveOwnerThread(current);
            return true;
        }
    } else if (current == getExclusiveOwnerThread()) {
        // 线程重入
        int nextc = c + acquires;
        if (nextc < 0) // overflow
            throw new Error("Maximum lock count exceeded");
        // state状态+1, 用于重入的线程在unlock时做-1操作
        setState(nextc);
        return true;
    }
    // 没有抢到锁
    // 1. 进来时, 锁就被占用了 c <> 0 && 当前未持有锁
    // 2. 进来时, c == 0, 但在争抢的过程中(compareAndSetState), 没有抢到
    return false;
}

// 这是AQS的方法
// 一个用 volatile 修饰的变量
// 实际的意义, 是在实现类中体现的
// 在ReentrantLock中, state表示, 是否持有锁
// 0: 表示没有线程占用这把锁
// 1: 表示有线程占用了这把锁
// >1: 表示这把锁重入的次数(n-1)
protected final int getState() {
    return state;
}


private transient volatile Node head;

private transient volatile Node tail;

private volatile int state;

private static final VarHandle HEAD;

static {

# 7. AQS.addWaiter()

private Node addWaiter(Node mode) {
    // Node 为双向链表结构
    // 设置为独占模式
    Node node = new Node(mode);
    for (;;) {
        // 拿到队列的尾节点
        Node oldTail = tail;
        if (oldTail != null) {
            // 把原来的尾节点, 做为当前node节点的前驱节点
            node.setPrevRelaxed(oldTail);
            // 尝试把当前node节点设置为队列的尾节点
            if (compareAndSetTail(oldTail, node)) {
                // 设置成功后, 原尾节点的后继指向当前node节点
                oldTail.next = node;
                return node;
            }
            // 未成功设置尾节点时, 就会重新进入for循环, 重新获取尾节点
        } else {
            // 初始化同步队列
            initializeSyncQueue();
        }
    }
}

// 初始化同步队列
private final void initializeSyncQueue() {
    Node h;
    // 队列的头节点和尾节点, 同时指向新创建的Node对象
    if (HEAD.compareAndSet(this, null, (h = new Node()))) {
        tail = h;
    }
}

// 通过CAS的方式, 设置队列尾节点
private final boolean compareAndSetTail(Node expect, Node update) {
    return TAIL.compareAndSet(this, expect, update);
}

# 8. AQS.acquireQueued()

// 从队列中获取到锁的使用权
final boolean acquireQueued(final Node node, int arg) {
    boolean interrupted = false;
    try {
        for (;;) {
            // 拿前node的前驱节点
            final Node p = node.predecessor();
            // 判断node前驱节点是不是队列中的第一个
            // 如果是队列中的第一个, 尝试获得锁
            if (p == head && tryAcquire(arg)) {
                setHead(node);
                p.next = null; // help GC
                return interrupted;
            }
            // 当p有后继节点时, 设置p.waitStatus = -1
            // 并从队列中踢出取消状态的节点
            if (shouldParkAfterFailedAcquire(p, node))
                interrupted |= parkAndCheckInterrupt();
        }
    } catch (Throwable t) {
        cancelAcquire(node);
        if (interrupted)
            selfInterrupt();
        throw t;
    }
}

private void setHead(Node node) {
    // node 设置为头节点
    head = node;
    // 当前节点中所包装的线程, 设置为空(help GC)
    node.thread = null;
    // 把上一个节点, 移出队列.
    node.prev = null;
}

private static boolean shouldParkAfterFailedAcquire(Node pred, Node node) {
    // 前驱节点的状态
    int ws = pred.waitStatus;
    // 如果前驱节点, 处于阻塞状态
    if (ws == Node.SIGNAL)
        // 当前节点, 也会处于阻塞状态
        return true;
    if (ws > 0) {
        do {
            // ws > 0, 只有一种状态, 就是取消状态
            // 把node.prev踢出, 队列
            node.prev = pred = pred.prev;
        } while (pred.waitStatus > 0); // 循环踢
        pred.next = node;
    } else {
        // ws != -1 && ws != 1 的情况下(可能会是初始值: 0)
        // 尝试设置前驱节点为阻塞阻塞
        pred.compareAndSetWaitStatus(ws, Node.SIGNAL);
    }
    return false;
}

private final boolean parkAndCheckInterrupt() {
    // 阻塞当前线程, 等待唤醒
    LockSupport.park(this);
    // 唤醒后, 先检查线程是否被打断
    return Thread.interrupted();
}

# 公平锁

上面看到的是, 非公平锁的实现方法. 下面来看一下公平锁的实现方法

# 2. FairSync.tryAcquire()

protected final boolean tryAcquire(int acquires) {
    final Thread current = Thread.currentThread();
    int c = getState();
    if (c == 0) {
        // 与nonfairTryAcquire() 貌似只差了hasQueuedPredecessors()一个方法
        // 判断是否有节点在排队
        // 如果有人在排队, 直接就去排队了...抢都不抢的
        if (!hasQueuedPredecessors() &&
            compareAndSetState(0, acquires)) {
            setExclusiveOwnerThread(current);
            return true;
        }
    }
    else if (current == getExclusiveOwnerThread()) {
        int nextc = c + acquires;
        if (nextc < 0)
            throw new Error("Maximum lock count exceeded");
        setState(nextc);
        return true;
    }
    return false;
}


public final boolean hasQueuedPredecessors() {
    Node h, s;
    // 判断队列头节点, 是否为空
    // 为空说明, 队列是空的, 直接返回False
    if ((h = head) != null) {
        // 头节点没有后继节点或后继节是取消状态时
        if ((s = h.next) == null || s.waitStatus > 0) {
            s = null; // traverse in case of concurrent cancellation
            // 从后往前, 依次从队列中踢出取消的节点
            for (Node p = tail; p != h && p != null; p = p.prev) {
                if (p.waitStatus <= 0) {
                    s = p; // s = head后面最近的非取消状态的节点
                }
            }
        }
        // 如果s存在, 且不是当前线程, 说明队列中有其他节点在排队
        if (s != null && s.thread != Thread.currentThread())
            return true;
    }
    return false;
}

# 1. AQS.release()

public final boolean release(int arg) {
    // 尝试释放锁
    if (tryRelease(arg)) {
        // 成功释放锁后, 拿到队列中的头节点
        Node h = head;
        // 通过上面shouldParkAfterFailedAcquire方法可知, 只有尾节点的waitStatus=0
        if (h != null && h.waitStatus != 0)
            // 把haed踢出队列 
            unparkSuccessor(h);
        return true;
    }
    return false;
}

# 2. Sync.tryRelease()

protected final boolean tryRelease(int releases) {
    // 持锁的临时状态
    int c = getState() - releases;
    // 检查当前线程, 是否为锁的独享线程
    if (Thread.currentThread() != getExclusiveOwnerThread())
        // 不是直接异常
        throw new IllegalMonitorStateException();
    // 判断是否重入锁的标识
    boolean free = false;
    if (c == 0) {
        // 如果不是重入锁, 释放独享线程
        free = true;
        setExclusiveOwnerThread(null);
    }
    // 刷新持锁状态
    setState(c);
    return free;
}

# 4. AQS.unparkSuccessor()

private void unparkSuccessor(Node node) {
    int ws = node.waitStatus;
    if (ws < 0) {
        node.compareAndSetWaitStatus(ws, 0);
    }

    Node s = node.next;
    // 没有后继, 或者后继为取消状态
    if (s == null || s.waitStatus > 0) {
        s = null;
      	// 从后往前, 依次从队列中踢出取消的节点
        for (Node p = tail; p != node && p != null; p = p.prev)
            if (p.waitStatus <= 0) {
                // s = node后面最近的非取消状态的节点
                s = p;
            }
    }
    // 叫醒后继节点的线程
    if (s != null) {
        LockSupport.unpark(s.thread);
    }
}
