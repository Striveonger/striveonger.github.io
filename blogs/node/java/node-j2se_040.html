package xyz.striveonger.demo.concurrent.chapter_03.juc.lock;

import com.google.common.collect.Lists;
import xyz.striveonger.demo.concurrent.chapter_03.thread.SleepHelper;

import java.util.List;
import java.util.concurrent.locks.Lock;
import java.util.concurrent.locks.ReadWriteLock;
import java.util.concurrent.locks.ReentrantLock;
import java.util.concurrent.locks.ReentrantReadWriteLock;

/**
 * @author Mr.Lee
 * @description:
 * @date 2021-04-20 9:56 PM
 */
public class ReadWriteLockExample {

    private Lock lock = new ReentrantLock();

    private volatile int x = 0;

    // 可重入的读写锁
    private ReadWriteLock readWriteLock = new ReentrantReadWriteLock(true);
    private Lock readLock = readWriteLock.readLock();
    private Lock writeLock = readWriteLock.writeLock();

    private Runnable read = () -> {
        try {
            readLock.lock();
            SleepHelper.sleepSeconds(1);
            String name = Thread.currentThread().getName();
            System.out.println(name + ", read x: " + x);
        } finally {
            readLock.unlock();
        }
    };

    private Runnable write = () -> {
        try {
            writeLock.lock();
            SleepHelper.sleepSeconds(1);
            x++;
            String name = Thread.currentThread().getName();
            System.out.println(name + ", write x: " + x);
        } finally {
            writeLock.unlock();
        }
    };



    public static void main(String[] args) {
        ReadWriteLockExample example = new ReadWriteLockExample();
        List<Thread> threads = Lists.newArrayList();
        for (int i = 0; i < 8; i++) {
            threads.add(new Thread(example.read, "t" + i));
        }
        for (int i = 0; i < 2; i++) {
            threads.add(new Thread(example.write, "t" + (i + 8)));
        }

        threads.forEach(Thread::start);

    }
}

# 3. Sync.tryAcquireShared()
// 尝试获取到分享锁
//  1: 成功拿到
// -1: 未拿到
protected final int tryAcquireShared(int unused) {
    Thread current = Thread.currentThread();
    // 获得当前锁的状态
    int c = getState();
    // 判断当前锁, 是否被排他锁占用
    if (exclusiveCount(c) != 0 &&
        // 当锁为独占锁时, 判断是否为重入
        getExclusiveOwnerThread() != current) {
        return -1;
    }
    // 读线程重入次数
    int r = sharedCount(c); 
    // 检查读线程是否要阻塞
    if (!readerShouldBlock() && 
        // 当前读锁重入次数是否超过最大值
        r < MAX_COUNT && 
        // CAS(r, r+1) 重入次数+1. CAS失败会调用fullTryAcquireShared
        compareAndSetState(c, c + SHARED_UNIT)) { 
        // 如果是第一个读的线程
        if (r == 0) {
            // 表示第一次读
            // 记录读线程对象
            firstReader = current;
            // 初始化第一个读线程读的次数
            firstReaderHoldCount = 1;
        } else if (firstReader == current) {
            // 刷新第一个读线程读的次数
            firstReaderHoldCount++;
        } else {
            // 实际就是一个ThreadLocal管理读的次数, 与firstReader作用一样
            HoldCounter rh = cachedHoldCounter;
            if (rh == null ||
                rh.tid != LockSupport.getThreadId(current)) {
                cachedHoldCounter = rh = readHolds.get();
            } else if (rh.count == 0) {
                readHolds.set(rh);
            }
            rh.count++;
        }
        return 1;
    }
    // 防止tryAcquireShared方法中CAS修改state失败
    // 循环的CAS修改state失败
    return fullTryAcquireShared(current);
}


// 这是Sync自己的常量与方法
// 共享锁的偏移量
// 共享锁与排他锁是通过AQS中state的32位二进制数表示的
// 高16位用来表示共享锁
// 低16位用来表示排他锁
final int SHARED_SHIFT   = 16;
final int SHARED_UNIT    = (1 << SHARED_SHIFT);
final int MAX_COUNT      = (1 << SHARED_SHIFT) - 1;
final int EXCLUSIVE_MASK = (1 << SHARED_SHIFT) - 1;

int exclusiveCount(int currentState) { 
    // 判断当前AQS中锁的状态, 是否被排他锁占有
    return currentState & EXCLUSIVE_MASK; 
}

int sharedCount(int currentState) {
    // 向右偏移16位
    // 去掉低16位的数, 就是共享锁的占用数量了....妙呀
    return currentState >>> SHARED_SHIFT;
}

/**
 * 判断读线程是否要阻塞
 */
boolean readerShouldBlock() {
    return apparentlyFirstQueuedIsExclusive();
}
// 判断队列的头节点是否为共享节点
boolean apparentlyFirstQueuedIsExclusive() {
    Node h, s;
    return (h = head) != null && // 队列非空
        (s = h.next)  != null && // 后继节点非空
        !s.isShared()         && // 后继节点为独占模式
        s.thread != null;        // 后继节点线程非空
    // 检查持锁线程head后继节点s是否为独占模式(写锁), 若是则返回True. 结合tryAcquireShared的逻辑, 如果True意味着读线程会被挂起无法共享锁.
    // 当持锁的是读线程时, 跟随其后的是一个写线程, 那么再后面来的读线程是无法获取读锁的, 只有等待写线程执行完后, 才能竞争.
    // 这是JDK为了避免写线程过分饥渴, 而做出的策略, 但有坑点就是, 如果某一读线程执行时间过长, 甚至陷入死循环, 后续线程会无限期挂起, 严重程度堪比死锁. 为避免这种情况, 除了确保读线程不会有问题外, 尽量用tryLock, 超时我们可以做出响应.
}


// 对tryAcquireShared方法的补充
// 防止tryAcquireShared方法中CAS修改state失败
int fullTryAcquireShared(Thread current) {
    HoldCounter rh = null;
    for (;;) {
        int c = getState();
        // 判断当前锁是否由写锁持有
        if (exclusiveCount(c) != 0) {
            // 判断是否为线程重入
            if (getExclusiveOwnerThread() != current) {
                // 直接返回失败
                return -1;
            }
        } else if (readerShouldBlock()) { // 检查读线程是否要阻塞
            // 头节点的后继节点为独占锁时, 需要当前线程阻塞
            // 确保不是重入的读锁
            // 上面tryAcquireShared方法中, 初始化过firstReader
            if (firstReader == current) {
                // assert firstReaderHoldCount > 0;
            } else {
                // 计数器未初始化
                if (rh == null) {
                    // 初始化计数器
                    rh = cachedHoldCounter;
                    // 判断计数器是否合法
                    if (rh == null || rh.tid != getThreadId(current)) {
                        // 重新从当前线程中, 拿到计数器
                        rh = readHolds.get();
                        // 没有调用过该线程
                        if (rh.count == 0) {
                            // 如果计数器为0, 则放弃治疗
                            // 从计数器持有者中, 删除该计数器
                            readHolds.remove();
                        }
                    }
                }
                if (rh.count == 0) { 
                    // 当前线程的调用次数为0时, 返回失败进入等待队列
                    return -1;
                }
            }
        }
        // 判断是否超过读线程限额
        if (sharedCount(c) == MAX_COUNT) {
            throw new Error("Maximum lock count exceeded");
        }
        // 最重要的就是这一句....CAS操作, 持有共享锁的数量 +1 
        if (compareAndSetState(c, c + SHARED_UNIT)) { // CAS(r, r+1)
            // CAS 成功后, 计数器+1
            if (sharedCount(c) == 0) {
                firstReader = current;
                firstReaderHoldCount = 1;
            } else if (firstReader == current) {
                firstReaderHoldCount++;
            } else {
                if (rh == null)
                    rh = cachedHoldCounter;
                if (rh == null || rh.tid != getThreadId(current))
                    rh = readHolds.get();
                else if (rh.count == 0)
                    readHolds.set(rh);
                rh.count++;
                cachedHoldCounter = rh; // cache for release
            }
            return 1;
        }
    }
}

# 4. AQS.doAcquireShared()
// 尝试获取分享锁
private void doAcquireShared(int arg) {
    // 向队列尾部, 添加共享模式的节点
    final Node node = addWaiter(Node.SHARED);
    // 线程打断的标识
    boolean interrupted = false;
    try {
        for (;;) {
            // 拿到当前节点的前继节点
            final Node p = node.predecessor();
            // 如果前继节点, 是头节点
            if (p == head) {
                // 尝试拿到共享锁
                int r = tryAcquireShared(arg);
                // 1: 成功, -1: 失败
                if (r >= 0) {
                    // 当前节点成功获取共享锁后, 释放当前节点及后继的共享节点中的线程, 开始运行
                    setHeadAndPropagate(node, r);
                    p.next = null; // help GC
                    return;
                }
            }
            // 当p有后继节点时, 设置p.waitStatus = -1
            // 并从队列中踢出取消状态的节点
            if (shouldParkAfterFailedAcquire(p, node)) {
                // 线程唤醒后, 返回线程是否被打断
                interrupted |= parkAndCheckInterrupt();
            }   
        }
    } catch (Throwable t) {
        // 当前节点, 取消尝试获取共享锁
        cancelAcquire(node);
        throw t;
    } finally {
        if (interrupted) {
            // 打断当前线程
            selfInterrupt();
        }
    }
}

// 成功获取共享锁后, 处理节点状态
private void setHeadAndPropagate(Node node, int propagate) {
    Node h = head;
    // 重置头节点
    setHead(node);
    // propagate: 传播状态为1时, 表示成功获取到共享锁
    // h == null: 队列的头节点为空
    // h.waitStatus < 0 : 头节点处于阻塞状态
    if (propagate > 0 || h == null || h.waitStatus < 0 || (h = head) == null || h.waitStatus < 0) {
        // 当前节点, 成功获取到共享锁后
        Node s = node.next; // 拿到当前节点的后继节点
        // 判断后继节点, 是否合法
        // s == null 说明是最后一个
        // s.isShared() 说明是分享模式的节点
        // 这里, 有人要问了....为啥不是node.thread执行, 而node.next.thread执行呢
        // 因为head出事化时, head.thread = null
        // 所以, 执行doReleaseShared时, node.thread已经在执行了.
        if (s == null || s.isShared()) {
            // 使队列中第二个节点中的线程开始运行
            doReleaseShared();
        }
    }
}

private void setHead(Node node) {
    head = node;
    node.thread = null;
    node.prev = null;
}

// 释放共享锁的资源(使当前节点及后继的共享节点中的线程, 开始运行~)
private void doReleaseShared() {
    for (;;) {
        Node h = head;
        // 放心, 当前要unpark的肯定是head节点
        if (h != null && h != tail) {
            int ws = h.waitStatus;
            if (ws == Node.SIGNAL) {
                // 重置waitStatus
                if (!h.compareAndSetWaitStatus(Node.SIGNAL, 0)) {
                    // 重置失败, 那就重试呗
                    continue;
                }
                // unpark 后继节点中的线程
                unparkSuccessor(h);
            }
            // 如果ws=0, 说明是初始化状态的节点. 
            // 连续的读操作节点可以依次进入临界区
            else if (ws == 0 && !h.compareAndSetWaitStatus(0, Node.PROPAGATE)) {
                // 设置置失败, 那就重试呗
                continue;
            }
        }
        if (h == head) {
            break;
        }
    }
}
// unpark 队列头节点的后继节点
private void unparkSuccessor(Node node) {
    int ws = node.waitStatus;
    // 再多加一次的判断
    if (ws < 0) {
        // 重置状态
        node.compareAndSetWaitStatus(ws, 0);
    }

    Node s = node.next;
    // 如果后继节点不合法
    if (s == null || s.waitStatus > 0) {
        s = null;
        // 从队列中选择, 距离node最近的合法后继节点
        for (Node p = tail; p != node && p != null; p = p.prev) {
            // 
            if (p.waitStatus <= 0) {
                s = p;
            }
        }
    }
    if (s != null) {
        // 这才是重点~
        // 开始运行node节点中的线程
        LockSupport.unpark(s.thread);
    }
}

// 并从队列中踢出取消状态的节点
private static boolean shouldParkAfterFailedAcquire(Node pred, Node node) {
    int ws = pred.waitStatus;
    if (ws == Node.SIGNAL)
        return true;
    if (ws > 0) {
        do {
            // 从后往前, 依次踢出取消的节点
            node.prev = pred = pred.prev;
        } while (pred.waitStatus > 0);
        pred.next = node;
    } else {
        // 设置前继节点为阻塞状态
        pred.compareAndSetWaitStatus(ws, Node.SIGNAL);
    }
    return false;
}

// node节点, 不再竞争锁
private void cancelAcquire(Node node) {
    if (node == null) {
        return;
    }
    node.thread = null;
    // 跳过node节点之前所有取消状态的节点
    Node pred = node.prev;
    while (pred.waitStatus > 0) {
        node.prev = pred = pred.prev;
    }

    
    Node predNext = pred.next;// node的副本指针
    // 设置node节点的状态为取消状态
    node.waitStatus = Node.CANCELLED;
    if (node == tail && compareAndSetTail(node, pred)) {
        // 如果node为尾节点, 直接把node踢出队列
        pred.compareAndSetNext(predNext, null);
    } else {
        int ws;
        if (pred != head &&  
            // 前继节点, 为阻塞状态 或者初始化Node.SIGNAL状态, 失败时.
            ((ws = pred.waitStatus) == Node.SIGNAL || (ws <= 0 && pred.compareAndSetWaitStatus(ws, Node.SIGNAL))) &&
            // 前继节点, 未初始化
            pred.thread != null) {
            Node next = node.next;
            if (next != null && next.waitStatus <= 0) {
                // 从队列中, 踢出node节点
                pred.compareAndSetNext(predNext, next);
            }
        } else {
            // unpark 后继节点
            unparkSuccessor(node);
        }
        node.next = node; // help GC
    }
}

# 3. Sync.tryReleaseShared()
// 尝试释放共享锁
protected final boolean tryReleaseShared(int unused) {
    // 当前线程的锁的持有计数器 -1
    Thread current = Thread.currentThread();
    if (firstReader == current) {
        // assert firstReaderHoldCount > 0;
        if (firstReaderHoldCount == 1)
            firstReader = null;
        else
            firstReaderHoldCount--;
    } else {
        HoldCounter rh = cachedHoldCounter;
        if (rh == null ||
            rh.tid != LockSupport.getThreadId(current))
            rh = readHolds.get();
        int count = rh.count;
        if (count <= 1) {
            readHolds.remove();
            if (count <= 0)
                throw unmatchedUnlockException();
        }
        --rh.count;
    }
    // 自旋CAS
    for (;;) {
        // 获得当前锁的状态
        int c = getState();
        // 相当于, 高16位的数 -1; 也就是共享锁持有数量-1
        int nextc = c - SHARED_UNIT;
        if (compareAndSetState(c, nextc)) { 
            // 理论上, 共享锁和排他锁不会同时工作的
            // 这句的意思就是, 当没有线程使用共享锁时, 返回true
            return nextc == 0;
        }
    }
}

# 4. AQS.doReleaseShared()
// 当没有线程持有锁时, 才会执行这个方法
// 使队列中等待的等待的线程, 开始运行
// 可能是排他锁的线程在阻塞, 也有可能是超出的共享线程((1<<16)个)的等待线程
private void doReleaseShared() {
    for (;;) {
        // 拿到队列的头节点
        Node h = head;
        // 判断头节点是否为空, 或者等于尾节点
        if (h != null && h != tail) {
            int ws = h.waitStatus;
            if (ws == Node.SIGNAL) { 
                // 如果头节点的后继节点的状态, 
                if (!h.compareAndSetWaitStatus(Node.SIGNAL, 0)) {
                    continue; // 重置失败, 那就重试呗
                }
                // unpark 头节点的后继节点
                unparkSuccessor(h);
            } 
            // 如果ws=0, 说明是初始化状态的节点. 
            // 连续的读操作节点可以依次进入临界区
            else if (ws == 0 && !h.compareAndSetWaitStatus(0, Node.PROPAGATE)) {
                continue; // 重置失败, 那就重试呗
            }
        }
        if (h == head) {
            // 跳出循环
            break;
        }
    }
}

# 3. Sync.tryAcquire()
protected final boolean tryAcquire(int acquires) {
    // 获取当前线程状态
    Thread current = Thread.currentThread();
    int c = getState();
    //  c & ((1 << 16) - 1) 当前排他锁的使用数量
    int w = exclusiveCount(c);
	// 当前线程的状态, 
    if (c != 0) {
        // c != 0 说明有线程持有锁(可能是共享锁哦~)
        // 当前排他锁的使用数量为0, 或者当前线程不是持有排他锁的线程, 返回Fasle
        // 排他锁的使用数量为0 是因为现在是共享锁的使用时间
        // 至于第二个条件....都告诉你是'排他锁'了呀
        if (w == 0 || current != getExclusiveOwnerThread()) {
            return false;
        }
        // 判断是否超过使用的最大值.
        if (w + exclusiveCount(acquires) > MAX_COUNT) {
            throw new Error("Maximum lock count exceeded");
        }
        // 刷新锁状态
        setState(c + acquires);
        return true;
    }
    // 尝试争抢排他锁
    if (writerShouldBlock() || !compareAndSetState(c, c + acquires)) {
        // 争抢失败后, 会进入队列中等待
        return false;
    }
    // 争抢到锁, 设置当前线程持有这把锁
    setExclusiveOwnerThread(current);
    return true;
}


// 公平锁
boolean writerShouldBlock() {
    // 检查队列, 是否可以执行排他锁
    return hasQueuedPredecessors();
}

boolean hasQueuedPredecessors() {
    Node t = tail;
    Node h = head;
    Node s;
    return h != t && // 队列不是空的
        // 头节点的后继节点不是空的, 后继节点的线程,不是当前线程
        ((s = h.next) == null || s.thread != Thread.currentThread());
}

// 非公平锁
// 不检查队列中
boolean writerShouldBlock() {
    return false;
}

public final boolean release(int arg) {
    // 尝试释放排他锁
    if (tryRelease(arg)) {
        Node h = head;
        if (h != null && h.waitStatus != 0) {
            // unpark 队列中头节点的后继节点的线程

# 3. Sync.tryRelease()
protected final boolean tryRelease(int releases) {
    // 判断当前线程, 是否持有排他锁
    if (!isHeldExclusively()) {
        throw new IllegalMonitorStateException();
    }
    // 持有锁的数量 -1
    int nextc = getState() - releases;
    // 当持有锁的数量为0时, 彻底释放(有可能是重入的嘛~)
    boolean free = exclusiveCount(nextc) == 0;
    if (free) {
        // 放开这把锁
        setExclusiveOwnerThread(null);
    }
    // 刷新锁状态
    setState(nextc);
    // 返回当前线程, 是否依旧持有锁
    return free;
}

# 总结

ReentrantLock 和 ReentrantReadWriteLock 就放到一起说了…嘿嘿~

# 公平锁

尝试获取锁前, 先要检查等待队列…如果有线程排队, 就不尝试争抢锁了, 直接去队列排队

# 非公平锁

不检查等待队列, 先尝试争抢锁, 抢到了就开始用..抢不到, 就去队列中排队

# 共享锁

共享锁, 只判断当前线程数, 是不否超过阈值或当前锁属于排他锁执行…如果是, 则进入队列排队..如果不是, 则直接执行(共享锁持有数 +1)

# 排他锁

ReentrantLock 和 WriteLock 都属于排他锁…

锁是所持有锁的线程独有的, 在锁被占用时, 其他线程都要乖乖的去队列里等着….
