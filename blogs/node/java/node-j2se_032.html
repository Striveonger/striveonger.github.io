# 1.1 Example

/**
 * @author Mr.Lee
 * @description:
 * @date 2021-07-10 09:34:14
 */
public class FutureExample {

    public static void main(String[] args) throws ExecutionException, InterruptedException {

        // 创建线程池, 下面再详细说....
        ExecutorService pool = Executors.newCachedThreadPool();

        // 1. 开始无参的Runnable执行
        System.out.println("\n===========Non Param Runnable===========");
        Future<?> future = pool.submit(() -> {
            System.out.println("Start Future...");
        });
        System.out.println("Future: " + future.get());

        // 2. 开始有参的Runnable执行
        System.out.println("\n===========Have Param Runnable===========");
        // 说实话, 这个用的有点儿别扭呢...
        Param param = new Param();
        Runnable runnable = () -> {
            System.out.println("Hello Runnable...");
            param.setParam("method", "Runnable");
        };
        Future<Param> fr = pool.submit(runnable, param);
        System.out.println("Future: " + fr.get());
        System.out.println("Param: " + param);

        // 3. 开始有参的Callable执行
        System.out.println("\n===========Callable===========");
        Callable<Param> callable = () -> {
            Param res = new Param();
            System.out.println("Hello Callable...");
            res.setParam("method", "Callable");
            return res;
        };
        Future<Param> fc = pool.submit(callable);
        System.out.println(fc.isDone());
        // 会阻塞一秒
        System.out.println("Future: " + fc.get());
        System.out.println(fc.isDone());

        // 现在就关闭线程池
        pool.shutdownNow();
    }

    private static class Param {

        private final Map<String, Object> args = new ConcurrentHashMap<>();

        public void setParam(String key, Object value) {
            args.put(key, value);
        }

        @Override
        public String toString() {
            return args.toString();
        }
    }
}

  Future: {method=Runnable}
  Param: {method=Runnable}
  
  ===========Callable===========
  false
  Hello Callable...
  Future: {method=Callable}

# 1.2 执行者

# 1.3 Runnable

无返回结果的可执行任务

Runnable本身是没有返回值的. 当Runnable做为, 执行任务放到执行器中执行时:

  • 在submit时, 不指定参数时, Future是的返回值就null
  • 在submit时, 指定参数时, 在执行Future时, 返回的是入参变量的引用. 实际上是在Runnable内, 对入参做了修改.

可以看到, 最后, 还是会被封装成一个Callable

# 1.5 FutureTask

# 1.6 CompletableFuture

用管理多个Future的结果, 也可产生多个异步任务...比较灵活.....(先知道有这么个东西, 和大致用处. 用的时候能想起来, 再翻开发文档也来的及).

/**
 * @author Mr.Lee
 * @description: 多个线程的结果聚合
 * @date 2021-07-10 16:10:56
 */
public class CompletableFutureExample {

    public static void main(String[] args) throws Exception {

        long start = 0, end = 0;
        String res = "";

        CompletableFuture<String> futureA = CompletableFuture.supplyAsync(() -> {sleep(1000); return "A";});
        CompletableFuture<String> futureB = CompletableFuture.supplyAsync(() -> {sleep(200); return "B";});
        CompletableFuture<String> futureC = CompletableFuture.supplyAsync(() -> {sleep(500); return "C";});
        start = System.currentTimeMillis();

        // 1. 原来的调用方法
//        String a = futureA.get();
//        String b = futureB.get();
//        String c = futureC.get();
//        res = a + b + c;

        // 2. 利用 CompletableFuture 过
        CompletableFuture.allOf(futureA, futureB, futureC).join();

        end = System.currentTimeMillis();
        System.out.println("Use Time: " + (end - start));
        System.out.println("Result: " + res);
        System.in.read();
    }

    private static void sleep(int timeout) {
        SleepHelper.sleepMilliSeconds(timeout);
        System.out.println(Thread.currentThread().getName() + ", Sleep After...");
    }
}

# 2.1.2 线程工厂

// 线程池默认的线程工厂, 我也可以实现 ThreadFactory, 来创建自己的线程工厂
private static class DefaultThreadFactory implements ThreadFactory {
    private static final AtomicInteger poolNumber = new AtomicInteger(1);
    private final ThreadGroup group;
    private final AtomicInteger threadNumber = new AtomicInteger(1);
    private final String namePrefix;

    DefaultThreadFactory() {
        SecurityManager s = System.getSecurityManager();
        group = (s != null) ? s.getThreadGroup() :
                                Thread.currentThread().getThreadGroup();
        namePrefix = "pool-" +
                        poolNumber.getAndIncrement() +
                        "-thread-";
    }

    public Thread newThread(Runnable r) {
        Thread t = new Thread(group, r,
                                namePrefix + threadNumber.getAndIncrement(),
                                0);
        // 一定不是守护线程
        if (t.isDaemon())
            t.setDaemon(false);
        // 取消线程优先级
        if (t.getPriority() != Thread.NORM_PRIORITY)
            t.setPriority(Thread.NORM_PRIORITY);
        return t;
    }
}

# 2.1.3 拒绝策略

  • AbortPolicy 直接抛异常
  • DiscardPolicy 丢掉当前任务, 不处理
  • DiscardOldestPolicy 任务等待队列take , 丢掉任务等待最久的那个任务, 新任务入队.
  • CallerRunsPolicy 调用者执行

一般都会选择, 自定义拒绝规则, 放到MQ或者落盘. 由定时任务来处理(降级处理的一种吧).

这就要根据具体业务, 具体分析了.

# 2.1.4 Example

/**
 * @author Mr.Lee
 * @description:
 * @date 2021-07-11 19:36:48
 */
public class ThreadPoolExecutorExample {

    public static void main(String[] args) {
        ExecutorService service = new ThreadPoolExecutor(
                2, 4,
                60, TimeUnit.SECONDS,
                new LinkedBlockingQueue<>(4),
                Executors.defaultThreadFactory(),
                new ThreadPoolExecutor.AbortPolicy());

        ThreadPoolExecutor executor = (ThreadPoolExecutor) service;
        System.out.println("Pool Size: " + executor.getPoolSize());
        System.out.println("Queue Size: " + executor.getQueue().size());
        System.out.println("Core Pool Size: " + executor.getCorePoolSize()); // 返回对象的常量

        System.out.println("Pool Start...");
        for (int i = 0; i < 8; i++) {
            service.submit(new Task(i));
        }
        try {
            // 这里会报错的哦~
            service.submit(new Task(9));
        } catch (Exception e) {
            e.printStackTrace();
        }

        BlockingQueue<Runnable> queue = executor.getQueue();
        for (Runnable runnable: queue) {
            System.out.println(runnable);
        }

        System.out.println("Pool Size: " + executor.getPoolSize());
        System.out.println("Queue Size: " + executor.getQueue().size());
        System.out.println("Core Pool Size: " + executor.getCorePoolSize());

    }

    private static class Task implements Runnable {
        private int index;

        public Task(int index) {
            this.index = index;
        }

        @Override
        public String toString() {
            return "Task{" +
                    "index=" + index +
                    '}';
        }

        @Override
        public void run() {
            System.out.println(String.format("%s, Task %d, Start...", Thread.currentThread().getName(), this.index));
            // 使线程阻塞
            try { System.in.read(); } catch (IOException ignored) { }
            System.out.println(String.format("%s, Task %d, End...", Thread.currentThread().getName(), this.index));
        }
    }
}

# 行为分析
  • 初始化后, 线程池中的线程数为0
  • 当有任务(Task)进来时, 判断当前线程数如果小于核心线程数, 则创建核心线程, 执行任务
  • 核心线程满了, 任务会进入任务队列(Queue).
  • 当等待队列满了之后, 创建非核心线程来该执行任务.
  • 当队列与线程都被沾满时, 执行拒绝策略

# 2.2 ForkJoinPool

ForkJoinPool 不是为了替代 ExecutorService, 而是它的补充, 在某些应用场景下性能比 ExecutorService 更好.

ForkJoinPool 主要用于实现“分而治之”的算法, 特别是分治之后递归调用的函数. 适合计算密集型的任务.

                    TimeUnit unit)

ForkJoinPool支持两种模式：

  1. 同步模式（默认方式）
  2. 异步模式

mode = asyncMode ? FIFO_QUEUE : LIFO_QUEUE

工作线程优先处理自身队列中的任务(LIFO或FIFO顺序, 由线程池构造时的参数 mode 决定), 自身队列为空时, 以FIFO的顺序随机窃取其它队列中的任务.

相较于ThreadPoolExecutor 的所有工作线程, 共享同一个任务队列的情况, ForkJoinPool 的每个工作线程都有自己的任务队列.

public class ForkJoinWorkerThread extends Thread {
    
    final ForkJoinPool pool;                    // 该工作线程归属的线程池
    final ForkJoinPool.WorkQueue workQueue;     // 对应的任务队列
 
    protected ForkJoinWorkerThread(ForkJoinPool pool) {
        super("aForkJoinWorkerThread");         // 指定工作线程名称
        this.pool = pool;
        this.workQueue = pool.registerWorker(this); 
    }
  
    // ...
}

# 2.2.2 Example

/**
 * @description:
 * @author: Mr.Lee
 * @date: 2021-07-13 17:10
 */
public class ForkJoinPoolExample {

    public static void main(String[] args) {
        ForkJoinPool executor = new ForkJoinPool();
        long[] numbers = LongStream.rangeClosed(1, 100000000).toArray();
        long start = System.currentTimeMillis();
        ForkJoinTask<Long> task = new SumTask(numbers, 0, numbers.length - 1);
        new Thread(() -> {SleepHelper.sleepMilliSeconds(1); System.out.println("当前活跃线程数: " + executor.getActiveThreadCount());}).start();
        Long result = executor.invoke(task);
//        SleepHelper.sleepMilliSeconds(1000);
        executor.shutdown();
//        long result = 0L;
//        for (int i = 0; i < numbers.length; i++) {
//            result += numbers[i];
//        }
        long end = System.currentTimeMillis();
        System.out.println("最终结果: " + result);
        System.out.println("活跃线程数: " + executor.getActiveThreadCount());
        System.out.println("窃取任务数: " + executor.getStealCount());
        System.out.println("最终耗时: " + (end - start));
    }

    private static class SumTask extends RecursiveTask<Long> {

        private final long[] numbers;

        private final int from, to;

        /**
         * 定义任务范围
         */
        public SumTask(long[] numbers, int from, int to) {
            this.numbers = numbers;
            this.from = from;
            this.to = to;
        }

        @Override
        protected Long compute() {
            // 控制任务粒度
            if (to - from <= 1000) {
                // 执行任务
                long total = 0;
                for (int i = from; i <= to; i++) {
                    total += numbers[i];
                }
                return total;
            } else {
                // 向下细分任务
                int middle = (from + to) / 2;
                SumTask left = new SumTask(numbers, from, middle);
                SumTask right = new SumTask(numbers, middle + 1, to);
                // 任务分解
                left.fork();
                right.fork();
                // 聚合任务结果
                return left.join() + right.join();
            }
        }
    }
}

   当前活跃线程数: 5
   最终结果: 5000000050000000
   活跃线程数: 0
   窃取任务数: 19
   最终耗时: 139
 */

# 3.3 newSingleThreadExecutor

创建一个 Executor, 它使用单个工作线程在无界队列中运行

最多一个线程在工作, 等待队列中可以有2^31 -1个任务

也就是说, 帮我们实现了一个通过队列实现任务调度的简单模型.

还有还有 ,就是帮我们维护了完整的线程生命周期

public static ExecutorService newSingleThreadExecutor() {
    return new FinalizableDelegatedExecutorService
        (new ThreadPoolExecutor(1, 1,
                                0L, TimeUnit.MILLISECONDS,
                                new LinkedBlockingQueue<Runnable>()));
}


/**
 * 可以将 FinalizableDelegatedExecutorService 视作一个代理类
 * 只有在 JVM GC 时, 会调用 关闭线程池的方法
 */
private static class FinalizableDelegatedExecutorService extends DelegatedExecutorService {
    FinalizableDelegatedExecutorService(ExecutorService executor) {
        super(executor);
    }

    protected void finalize() {
        super.shutdown();
    }
}

public static void testScheduledThreadPool() throws InterruptedException {
    ScheduledExecutorService service = Executors.newScheduledThreadPool(2);
    // 测试延迟执行
    CountDownLatch count = new CountDownLatch(20);
    for (int i = 0; i < 10; i++) {
        int finalI = i;
        Object lock = new Object();
        service.schedule(() -> {
            synchronized (lock) {
                SleepHelper.sleepMilliSeconds(1000);
                System.out.printf("Index: %d, Time: %s%n", finalI, DateUtil.format(LocalDateTime.now(), "HH:mm:ss.SSS"));
                count.countDown();
            }
        }, 1, TimeUnit.SECONDS);
    }

# 3.5 newSingleThreadScheduledExecutor

创建一个单线程执行器, 它可以安排命令在给定的延迟后运行, 或定期执行

public static ScheduledExecutorService newSingleThreadScheduledExecutor() {
    return new DelegatedScheduledExecutorService(new ScheduledThreadPoolExecutor(1));
}

/**
 * 就可理解成一个, 包装类
 * 对DelegatedExecutorService的包装
 */
private static class DelegatedScheduledExecutorService extends DelegatedExecutorService implements ScheduledExecutorService {
    private final ScheduledExecutorService e;
    DelegatedScheduledExecutorService(ScheduledExecutorService executor) {
        super(executor);
        e = executor;
    }
    
    // 执行定期任务
    public ScheduledFuture<?> schedule(Runnable command, long delay, TimeUnit unit) {
        return e.schedule(command, delay, unit);
    }
    public <V> ScheduledFuture<V> schedule(Callable<V> callable, long delay, TimeUnit unit) {
        return e.schedule(callable, delay, unit);
    }
    public ScheduledFuture<?> scheduleAtFixedRate(Runnable command, long initialDelay, long period, TimeUnit unit) {
        return e.scheduleAtFixedRate(command, initialDelay, period, unit);
    }
    public ScheduledFuture<?> scheduleWithFixedDelay(Runnable command, long initialDelay, long delay, TimeUnit unit) {
        return e.scheduleWithFixedDelay(command, initialDelay, delay, unit);
    }
}
