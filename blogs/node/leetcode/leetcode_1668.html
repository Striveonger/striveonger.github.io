LeetCode-Python--1668.最大重复子字符串

# 暴力破解

#!/usr/bin/python3

class Solution:
    def maxRepeating(self, sequence: str, word: str) -> int:
        sum, i, j = 0, 0, 0
        list = []
        while i < len(sequence) and j < len(word):
            if sequence[i] == word[j]:
                # 匹配到相同字符, 指针同时后移
                i += 1
                j += 1
            else:
                if sum > 0:
                    list.append(sum)
                if j == 0:
                    i += 1
                else:
                    # sequence 这一轮, 只向后移一位
                    i -= (j - 1)
                # 字符不匹配, word指针回到原点
                j = 0
                sum = 0

            if j == len(word):
                sum += 1
                j = 0
        list.append(sum)
        return max(list)

class Solution:
    def maxRepeating(self, sequence: str, word: str) -> int:
        res = 0
        while word * (res + 1) in sequence:
            res += 1
        return res
