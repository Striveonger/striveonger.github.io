# Graph

自己熟悉的处理结构

/**
 * @author Mr.Lee
 * @description: 图(由点和边组成)
 * @date 2021-08-07 15:01:39
 */
public class Graph<T> {

    public Map<T, Node<T>> nodes;
    public Set<Edge<T>> edges;

    public Graph() {
        nodes = new HashMap<>();
        edges = new HashSet<>();
    }
}

/**
 * @author Mr.Lee
 * @description: 图中的点
 * @date 2021-08-07 15:01:58
 */
public class Node<T> {

    /**
     * 点的值
     */
    public T value;

    /**
     * 点的入度
     */
    public int in;

    /**
     * 点的出度
     */
    public int out;

    /**
     * 点的所有邻居
     */
    public List<Node<T>> nexts;

    /**
     * 点的所有边
     */
    public List<Edge<T>> edges;

    public Node(T value) {
        this.value = value;
        this.in = 0;
        this.out = 0;
        this.nexts = new ArrayList<>();
        this.edges = new ArrayList<>();
    }
}

/**
 * @author Mr.Lee
 * @description: 图中的边
 * @date 2021-08-07 15:02:10
 */
public class Edge<T> {

    /**
     * 边的权重
     */
    public int weight;

    /**
     * 边的起点
     */
    public Node<T> from;

    /**
     * 边的终点
     */
    public Node<T> to;

    public Edge(int weight, Node<T> from, Node<T> to) {
        this.weight = weight;
        this.from = from;
        this.to = to;
    }
}

# Adapter

可能题目中对图的描述方式, 和我们自己的不太一样, 所以要写一个适配器

# AdjacencyTable

经典的邻接表法的适配器

/**
 * @author Mr.Lee
 * @description:
 * @date 2021-08-08 15:03:08
 */
public interface Adapter<R, T> {

    Graph<R> createGraph(T data);

    default void updateGraph(Graph<R> graph, Node<R> fromNode, Node<R> toNode) {
        // 向from添加邻居节点
        fromNode.nexts.add(toNode);
        // 设置from的直接边
        Edge<R> edge = new Edge<>(0, fromNode, toNode);
        fromNode.edges.add(edge);
        graph.edges.add(edge);

        // 更新入度与出度
        fromNode.out++;
        toNode.in++;
    }
}

/**
 * @author Mr.Lee
 * @description: 邻接表法表示的图, 转换成自己的表示方式
 * @date 2021-08-08 15:59:31
 */
public class AdjacencyTableAdapter implements Adapter<String, Map<String, Set<String>>> {

    @Override
    public Graph<String> createGraph(Map<String, Set<String>> data) {
        Graph<String> graph = new Graph<>();
        for (Map.Entry<String, Set<String>> entry: data.entrySet()) {
            String from = entry.getKey();
            if (!graph.nodes.containsKey(from)) {
                graph.nodes.put(from, new Node<>(from));
            }
            Node<String> fromNode = graph.nodes.get(from);
            for (String to : entry.getValue()) {
                if (!graph.nodes.containsKey(to)) {
                    graph.nodes.put(to, new Node<>(to));
                }
                Node<String> toNode = graph.nodes.get(to);
                updateGraph(graph, fromNode, toNode);
            }
        }
        return graph;
    }
}

# LeetCode

以LeetCode 802. 找到最终的安全状态 为例.

输入: [[1,2],[2,3],[5],[0],[5],[],[]]

/**
 * @author Mr.Lee
 * @description:
 * @date 2021-08-08 15:01:46
 */
public class LeetCodeTopic802Adapter implements Adapter<Integer, Integer[][]> {

    /**
     * 1. 有向图
     * 2. 一维数组当前索引, 代表当前点
     * 3. 二维数组的values, 代表邻居节点
     * 4. 边, 没有权重
     */
    @Override
    public Graph<Integer> createGraph(Integer[][] data) {
        Graph<Integer> graph = new Graph<>();
        for (int i = 0; i < data.length; i++) {
            Integer from = i;
            if (!graph.nodes.containsKey(from)) {
                graph.nodes.put(from, new Node<>(from));
            }
            Node<Integer> fromNode = graph.nodes.get(from);
            Integer[] toNodes = data[i];
            for (Integer to : toNodes) {
                if (!graph.nodes.containsKey(to)) {
                    graph.nodes.put(to, new Node<>(to));
                }
                // 设置from的直接邻居
                Node<Integer> toNode = graph.nodes.get(to);
                updateGraph(graph, fromNode, toNode);
            }
        }
        return graph;
    }
}

/**
 * @author Mr.Lee
 * @description:
 * @date 2021-08-08 15:48:28
 */
public class PrintGraphExample {
    public static void main(String[] args) {
        AdjacencyTableAdapter adapter = new AdjacencyTableAdapter();
        Map<String, Set<String>> bfsData = new HashMap<>();
        bfsData.put("A", Stream.of("B", "C").collect(Collectors.toSet()));
        bfsData.put("B", Stream.of("K", "C").collect(Collectors.toSet()));
        bfsData.put("C", Stream.of("E", "F").collect(Collectors.toSet()));
        bfsData.put("K", Stream.of("E", "A").collect(Collectors.toSet()));
        bfsData.put("E", Stream.of("F").collect(Collectors.toSet()));
        bfsData.put("F", Collections.EMPTY_SET);
        Graph<String> bfsGraph = adapter.createGraph(bfsData);
        bfs(bfsGraph.nodes.get("A"));
    }

    private static void bfs(Node<String> start) {
        if (Objects.isNull(start)) {
            return;
        }
        Queue<Node<String>> queue = new LinkedList<>();
        Set<Node<String>> set = new HashSet<>();
        queue.add(start);
        while (!queue.isEmpty()) {
            Node<String> curr = queue.poll();
            if (set.contains(curr)) {
                continue;
            }
            set.add(curr);
            print(curr);
            for (Node<String> node : curr.nexts) {
                queue.add(node);
            }
        }
    }

    private static void print(Node<String> node) {
        System.out.println(node.value);
    }
}

# DFS

深度优先遍历

# 1. 创建图结构

用邻接表法创建

# 2. 算法描述

  • 假设从A节点开始.

  • A 节点入栈, 并注册A节点, 同时输出A节点

  • 开始弹栈 :

  1. A节点弹出, 寻找A节点的邻居节点
  2. B, C, k 选一个(假设选B), 检查B节点是否注册过
  3. 这时B节点是没有注册过的. 那么先把A节点压栈, 再把B节点压栈
  4. 注册B节点, 同时输出B节点, 然后不再往下找A节点的邻居了
  5. 继续弹栈 :
  6. B 节点弹出, 寻找B节点的邻居节点
  7. C, E 选一个(假设选C), 检查C节点是否注册过
  8. 这时C节点是没有注册过的. 那么先把B节点压栈, 再把C节点压栈
  9. 注册C节点, 同时输出C节点, 然后不再往下找B节点的邻居了
  10. 继续弹栈 :
  11. C 节点弹出, 寻找C节点的邻居节点
  12. D, 检查D节点是否注册过
  13. 这时D节点是没有注册过的. 那么先把C节点压栈, 再把D节点压栈
  14. 注册D节点, 同时输出D节点(此时, 栈里存储的, 就是从A到D的路径)
  15. 继续弹栈 :
  16. D 节点弹出, 寻找D节点的邻居节点
  17. 因为D节点没有邻居节点, 所以直接弹出
  18. 继续弹栈 :
  19. C 节点弹出, 寻找C节点的邻居节点
  20. D, 检查D节点是否注册过
  21. 这时D节点是已经注册过了, 那么不对D节点做任何操作
  22. C 节点只有一个邻居节点, 也已经注册过了, 所以C节点就被弹出了
  23. 继续弹栈 :
  24. B 节点弹出, 寻找B节点的邻居节点
  25. C, E 选一个(假设选C), 检查C节点是否注册过
  26. C节点, 已经注册过了, 不做处理.
  27. C, E, 选择E, 检查E节点是否注册过
  28. 这时E节点是没有注册过的. 那么先把B节点压栈, 再把E节点压栈
  29. 注册E节点, 同时输出E节点
  30. 继续弹栈 :
  31. E 节点弹出, 寻找E节点的邻居节点
  32. F, 检查F节点是否注册过
  33. 这时F节点是没有注册过的. 那么先把E节点压栈, 再把F节点压栈
  34. 注册F节点, 同时输出F节点
  35. 继续弹栈 :
  36. F 节点弹出, 寻找F节点的邻居节点
  37. C, 检查C节点是否注册过
  38. 这时C节点是已经注册过了, 那么不对C节点做任何操作
  39. F 节点只有一个邻居节点, 也已经注册过了, 所以F节点就被弹出了
  40. 继续弹栈 :
  41. E 节点弹出, 寻找E节点的邻居节点
  42. F, 检查F节点是否注册过
  43. 这时F节点是已经注册过了, 那么不对F节点做任何操作
  44. E 节点只有一个邻居节点, 也已经注册过了, 所以E节点就被弹出了
  45. 继续弹栈 :
  46. B 节点弹出, 寻找B节点的邻居节点
  47. C, E, 这时C, E节点都已经注册过了, 那么不做任何操作
  48. B节点, 也就被弹出了
  49. 继续弹栈 :
  50. A 节点弹出, 寻找A节点的邻居节点
  51. B, C, K (B, C 都被注册过, K 没有注册过), 先把A节点压栈, 再把K节点压栈
  52. 注册K节点, 同时输出K节点
  53. 继续弹栈 :
  54. K节点弹出, K节点没有邻居节点(所以跳过)
  55. 继续弹栈 :
  56. A节点弹出, A的所以邻居节点都注册过了, 跳过.
  57. 此时, 栈中就空了, 结束While循环

/**
 * @author Mr.Lee
 * @description:
 * @date 2021-08-08 15:48:28
 */
public class PrintGraphExample {

    public static void main(String[] args) {
        Map<String, Set<String>> dfsData = new HashMap<>();
        dfsData.put("A", Stream.of("B", "C", "K").collect(Collectors.toSet()));
        dfsData.put("B", Stream.of("C", "E").collect(Collectors.toSet()));
        dfsData.put("C", Stream.of("D").collect(Collectors.toSet()));
        dfsData.put("D", Stream.of("C").collect(Collectors.toSet()));
        dfsData.put("E", Stream.of("F").collect(Collectors.toSet()));
        dfsData.put("F", Stream.of("C").collect(Collectors.toSet()));
        dfsData.put("K", Collections.EMPTY_SET);
        Graph<String> dfsGraph = adapter.createGraph(dfsData);
        dfs(dfsGraph.nodes.get("A"));
    }

    private static void dfs(Node<String> start) {
        Deque<Node<String>> stack = new LinkedList<>();
        Set<Node<String>> set = new HashSet<>();
        // 入栈就注册
        stack.push(start);
        set.add(start);
        // 注册就打印
        print(start);
        while (!stack.isEmpty()) {
            Node<String> curr = stack.poll();
            for (Node<String> node : curr.nexts) {
                if (!set.contains(node)) {
                    stack.push(curr);
                    stack.push(node);
                    set.add(node);
                    print(node);
                    break;
                }
            }
        }
    }

    private static void dfs_error(Node<String> start) {
        Deque<Node<String>> stack = new ArrayDeque<>();
        Set<Node<String>> set = new HashSet<>();
        stack.push(start);
        while (!stack.isEmpty()) {
            Node<String> curr = stack.poll();
            if (set.contains(curr)) {
                continue;
            }
            set.add(curr);
            print(curr);
            boolean flag = true;
            for (Node<String> node : curr.nexts) {
                // 这里是将所有的邻居加入到了栈中(这不tm成了宽度优先遍历了嘛~)
                if (flag && !set.contains(node)) {
                    stack.push(curr);
                    stack.push(node);
                    flag = false;
                } else if (!set.contains(node)) {
                    stack.push(node);
                }
            }
        }
    }

    private static void print(Node<String> node) {
        System.out.println(node.value);
    }
}

# 3. 拓扑排序

一定是有向无环图, 才配的上拓扑排序哦~

拓扑序, 不是唯一的, 如上图, 可以表示为: A -> B -> C -> D, 也可以表示为: A -> C -> B -> D. 但是这两种都是对的

# TopologySort

尝试用BFS的方式, 求出下图的拓扑序.

# 1. 创建图结构

# 2. 算法描述

  • 遍历存储每个节点的入度信息, 并将入度为 0 的节点插入队列.

  • 依次消除队列里节点, 在图中的影响:
  1. B节点 出队, 打印, 并消除影响

因为消除了B节点 的影响后, 邻居节点的入度 都没有降到0, 所以没有向队列中加入新的节点

  1. A节点 出队, 打印, 并消除影响

消除了A节点 的影响后, 邻居C节点的入度 降到0, 所以向队列中加入C节点

  1. C节点 出队, 打印, 并消除影响

消除了C节点 的影响后, 邻居D节点的入度 降到0, 所以向队列中加入D节点

  1. D节点 出队, 打印, 并消除影响

消除了D节点 的影响后, 邻居E节点的入度 降到0, 所以向队列中加入E节点

  1. E节点 出队, 打印, 并消除影响

消除了E节点 的影响后, 邻居F节点的入度 降到0, 所以向队列中加入F节点

  1. F节点 出队, 打印, 并消除影响

F节点 没有邻居节点, 队列为空. 循环结束.

# 3. 具体实现

/**
 * @description: 拓扑排序
 * @author: Mr.Lee
 * @date: 2021-08-11 10:30
 */
public class TopologySortExample {

    public static void main(String[] args) {
        // 准备图数据
        AdjacencyTableAdapter adapter = new AdjacencyTableAdapter();
        Map<String, Set<String>> bfsData = new HashMap<>();
        bfsData.put("A", Stream.of("C").collect(Collectors.toSet()));
        bfsData.put("B", Stream.of("C", "D", "F").collect(Collectors.toSet()));
        bfsData.put("C", Stream.of("D", "E").collect(Collectors.toSet()));
        bfsData.put("D", Stream.of("E", "F").collect(Collectors.toSet()));
        bfsData.put("E", Stream.of("F").collect(Collectors.toSet()));
        bfsData.put("F", Collections.EMPTY_SET);
        Graph<String> graph = adapter.createGraph(bfsData);
        sort(graph);
    }

    private static void sort(Graph<String> graph) {
        // 必须是有向无环图哦~
        Queue<Node<String>> queue = new LinkedList<>();
        Map<Node<String>, Integer> nodeInMpa = new HashMap<>();
        // 1. 将图中, 所有的节点及节点入度插入到Map中
        for (Node<String> node : graph.nodes.values()) {
            nodeInMpa.put(node, node.in);
            // 2. 将入度为0的节点插入到队列中.
            if (node.in == 0) {
                queue.offer(node);
            }
        }
        while (!queue.isEmpty()) {
            Node<String> curr = queue.poll();
            System.out.println(curr.value);
            for (Node<String> node : curr.nexts) {
                int in = nodeInMpa.get(node) - 1;
                nodeInMpa.put(node, in);
                if(in == 0) {
                    queue.offer(node);
                }
            }
        }
    }
}
