public class FourArithmetic {

    /**
     * 加
     */
    public int add(int x, int y) {
        int ans = x;
        while (y != 0) {
            // 无进位相加
            ans = x ^ y;
            // 计算进位, 直到进位为0时, 结束
            y = (x & y) << 1;
            // 更新值
            x = ans;
        }
        return ans;
    }

    /**
     * 相反数

        // 补码, 相反数, 取反+1
        return add(~x, 1);

        return n < 0;

     * 减

        int a = isNeg(x) ? negNum(x) : x;
        int b = isNeg(y) ? negNum(y) : y;
        int ans = 0;
        for (int i = 30; i >= 0; i = minus(i, 1)) {
            // 当 a >= b << i时, 说明答案的i位, 一定为 1
            if ((a >> i) >= b) {
                // 答案的i位, 设为1
                ans |= 1 << i;

                a = minus(a, b << i);

        if (x == Integer.MIN_VALUE && y == Integer.MIN_VALUE) {

            }
        }
        return div(x, y);
    }

# 5.1 分割方法

/**
 * 分区
 * 小于arr[R]的放前面, 等于arr[R]放中间, 大于arr[R]放后面.
 * 小于区和大于区, 是无序的, 可以任意放
 */
private int[] partition(int[] arr, int L, int R) {
    // 小于区的右边界
    int lessR = L - 1;
    // 大于区的左边界
    int moreL = R;
    int index = L;
    while (moreL > index) {
        if (arr[index] > arr[R]) {
            // 如果当前数, 大于标致位的数, 当前数与大于区前一个数做交换, 大于区左扩
            swap(arr, index, --moreL);
        } else if (arr[index] < arr[R]) {
            // 如果当前数, 小于标致位的数, 当前数与小于区后一个数做交换, 小于区右扩, 当前数游标右移
            swap(arr, ++lessR, index++);
        } else {
            // 如果当前数, 等于标致位的数, 当前数游标右移
            index++;
        }
    }
    // 最后, 把标致位与大于区的
    swap(arr, moreL, R);
    return new int[]{lessR + 1, moreL};
}

class Task {
    public Task(int l, int r) {
        L = l;
        R = r;
    }

/**
 * 非递归版本

public void process(int[] arr) {
    Deque<Task> stack = new LinkedList<>();
    stack.addFirst(new Task(0, arr.length - 1));
    while (!stack.isEmpty()) {
        Task task = stack.pollFirst();
        int[] extent = partition(arr, task.L, task.R);
        if (extent[0] > task.L) {
            stack.addFirst(new Task(task.L, extent[0] - 1));
        }
        if (extent[1] < task.R) {
            stack.addFirst(new Task(extent[1] + 1, task.R));
        }
    }
}

# 6.1 合并方法

原理是通过双指针, 合并两个有序的数组.

L ~ M 和 M + 1 ~ R 两个区间范围, 被视为两个要合并的数组.

/**
 * 合并数组中两个有序的范围
 * 合并 L ~ M, M+1 ~ R区间, 使 L ~ R 范围内有序
 * 缺省: L ~ M 和 M+1 ~ R 连个范围内的数据, 是有序的
 */
private static void merge(int[] arr, int L, int M, int R) {
    int P1 = L, P2 = M + 1, i = 0;
    int[] help = new int[R - L + 1];
    // 双指针, 合并数组.
    while (P1 <= M && P2 <= R) {
        help[i++] = arr[P1] <= arr[P2] ? arr[P1++] : arr[P2++];
    }
    // 处理没有越界的分支
    while (P1 <= M) {
        help[i++] = arr[P1++];
    }

public void sort(int[] arr) {
    int step = 1;
    int N = arr.length;
    while (step < N) {
        int R = N - 1;
        while (R >= 0) {
            int M = Math.max(0, R - step);
            // 只有步长为1时, M和L都可能会是0, 否则, 就说明没有左分支
            if (M == 0 && step > 1) {
                break;
            }
            int L = Math.max(0, M - step + 1);
            merge(arr, L, M, R);
            R = L - 1;
        }
        if (step > N / 2) { // 控制边界
            // N/2 是向下取整的, 不能>=
            break;
        }
        step = step << 1;
    }
}

/**
 * 查找 arr中是否包含x

 */
private static boolean find(int arr[], int x) {
  if (Objects.isNull(arr) || arr.length < 1) {
    return false;
  }
  int L = 0, R = arr.length;
  while (L <= R) {
    int mid = (L + R) / 2;
    if (arr[mid] == x) {
      // 找到了, 要找的值
      return true;
    } else if (arr[mid] > x) {
      // 当中间位置上的数, 大于x时, mid以后的数就都不要了
      R = mid - 1;
    } else {
      // 当中间位置上的数, 小于x时, mid以前的数就都不要了
      L = mid + 1;
    }
  }
  return false;
}

/**
 * 查找数组峰值
 * 有一个隐藏条件: 相邻的两个数不等.
 */
public int findPeakElement(int[] nums) {
  if (Objects.isNull(nums) || nums.length == 0) {
    return -1;
  }
  if (nums.length == 1) {
    return 0;
  }
  int N = nums.length;
  if (nums[0] > nums[1]) {
    return 0;
  }
  if (nums[N - 2] < nums[N - 1]) {
    return N - 1;
  }
  // 如果代码走到这, 说明: 0 ~ 1是上涨趋势的, N-2 ~ N-1是下降趋势的. 
  // 因为相邻不相等的原则在, 所以必定有一个数处在"峰值"位置上.
  int L = 0, R = N - 1;
  while (L <= R) {
    int m = (L + R) / 2;
    if (nums[m] > nums[m - 1] && nums[m] > nums[m + 1]) {
      // 满足条件, 直接返回位置
      return m;
    } else if (nums[m] < nums[m - 1]) {
      // 说明: m ~ m+1 一定是下降趋势的. L ~ m 呈现: /  \ 趋势
      // 可以保证, L ~ m 里, 肯定有"峰值", 顾舍去 m 之后的数
      R = m;
    } else {
      // 说明: m-1 ~ m 一定是上涨趋势的. m ~ R 呈现: /  \ 趋势
      // 可以保证, m ~ R 里, 肯定有"峰值", 顾舍去 m 之前的数
      L = m;
    }
  }
  return -1;
}

            node.setNext(head);
            head.setPrev(node);
        } else {
            tail = node;
        }
        head = node;
        size++;
    }

        Node<V> node = new Node<>(item);
        if (Objects.nonNull(tail)) {
            node.setPrev(tail);
            tail.setNext(node);
        } else {
            head = node;
        }
        tail = node;
        size++;
    }

    public V pollFirst() {
        if (Objects.nonNull(head)) {
            Node<V> next = head.getNext();
            V result = head.getValue();
            if (Objects.nonNull(next)) {
                // 释放 head 指针指向的对象(要出队的对象)
                next.setPrev(null);
                // head 指针指向队列中的下一个节点
                head = next;
            } else {
                head = null;
                tail = null;
            }
            size--;
            return result;
        }
        return null;
    }

    public V pollLast() {
        if (Objects.nonNull(tail)) {
            V result = tail.getValue();
            Node<V> prev = tail.getPrev();
            if (Objects.nonNull(prev)) {
                // 一定要记得释放tail的指针.
                // 使 tail 不可达...JVM会自动清理不可达的对象
                // 释放 tail 指针指向的对象
                prev.setNext(null);
                // tail 指针向前移动一位
                tail = prev;
            } else {
                tail = null;
                head = null;
            }
            size--;
            return result;
        }
        return null;
    }

        if (Objects.nonNull(head)) {
            return head.getValue();

        if (Objects.nonNull(tail)) {
            return tail.getValue();
        }
        return null;

        return size;

        return size == 0;

        return head;

            v = v % 10;
            carry = 1;

        l2 = l2.next;
        if (Objects.nonNull(l1)) {
            last = l1;
        }

    // 2. L1链表还没有到头, 且还有进位
    while (Objects.nonNull(l1) && carry > 0) {

public class BitMap {

    private int offset;
    private long[] sets;

    /**
     * 默认从0开始计数
     *
     * @param max
     */
    public BitMap(int max) {
        this(0, max);
    }

    public BitMap(int min, int max) {
        if (min > max) {
            // 校验边界条件
            throw new IllegalArgumentException();
        }
        // 计算偏移量
        offset = -min;
        // 计算最大值
        max += offset;
        // 计算最大容量
        // (max + 64) >> 6 ==>> (max + 64) / 64 ==>> max / 64 + 1
        sets = new long[(max + 64) >> 6];
    }

    public void add(int num) {
        // 偏移
        num += offset;
        // num >> 6 ==>> num / 64 ==>> 计算num对应的索引位置
        // 1L << (num & 63) =>> 1 << (num % 64) ==>> num占位
        sets[num >> 6] |= 1L << (num & 63);
    }

    public void delete(int num) {
        num += offset;
        sets[num >> 6] &= ~(1L << (num & 63));
    }

    public boolean contains(int num) {
        num += offset;
        return (sets[num >> 6] & 1L << (num & 63)) != 0;
    }
}

public List<List<Integer>> levelOrderBottom(TreeNode root) {
    // 定义返回对象
    List<List<Integer>> ans = new LinkedList<>();
    // 处理边界条件
    if (Objects.nonNull(root)) {
        // 定义一个队列, 用来存放待遍历的节点
        Queue<TreeNode> queue = new LinkedList<>();
        // 加入根节点
        queue.offer(root);
        // 遍历, 直到队列空为止
        while (!queue.isEmpty()) {
            // 遍历一开始, 抓到本次需要处理的次数(因为下面, queue.size 是会变化的)
            int N = queue.size();
            // 定义当前的容器, 长度就是N
            List<Integer> curr = new ArrayList<>(N);
            for (int i = 0; i < N; i++) {
                // 节点出队
                TreeNode node = queue.poll();
                // 加入到当前容器中
                curr.add(node.val);
                // 如果有左节点, 加入队列
                if (Objects.nonNull(node.left)) {
                    queue.offer(node.left);
                }
                // 如果有右节点, 加入队列
                if (Objects.nonNull(node.right)) {
                    queue.offer(node.right);
                }
            }
            // 因为要逆序输出, 所以每次都插在0位置
            ans.add(0, curr);
        }
    }
    return ans;
}

/**
 * 制造TreeNode
 * @param pre 先序的数组
 * @param L1 先序数组开始位置索引
 * @param R1 先序数组结束位置索引
 * @param L2 中序数组开始位置索引
 * @param R2 中序数组结束位置索引
 * @param inValIndexMap 位置索引
 * @return
 */
private TreeNode make(int[] pre, int L1, int R1, int L2, int R2, Map<Integer, Integer> inValIndexMap) {
    // 控制边界
    if (L1 > R1) {
        // 处理单支没值的情况
        return null;
    }
    // 先序遍历的第一个元素, 是树的root节点
    TreeNode root = new TreeNode(pre[L1]);
    // 如果L1 == R1则, 只有一个节点, 往下没有节点了
    if (L1 < R1) {
        // 找到 root节点, 在中序遍历中的位置
        int index = inValIndexMap.get(pre[L1]);
        // 构建左树节点, (index - L2) 左树长度, (L1 + index - L2) 先序数组中, 左树的结束位置索引
        // 如果树只有右节点, 没有左节点, 那么 (L1 + index - L2) == L1, 所以
        root.left = make(pre, L1 + 1, L1 + index - L2, L2, index - 1, inValIndexMap);
        // 构建右树节点, (L1 + index - L2 + 1) 先序数据右树起点索引
        root.right = make(pre, L1 + index - L2 + 1, R1, index + 1, R2, inValIndexMap);
    }
    return root;
}

private Result make(TreeNode root) {
    if (Objects.isNull(root)) {
        // 空节点, 默认是平衡的, 且高度为 0
        return new Result(true, 0);
    }
    // 左节点的返回值
    Result left = make(root.left);
    // 右节点的返回值
    Result right = make(root.right);
    // 判断当前节点是否平衡
    boolean isBalanced = left.isBalanced && right.isBalanced && 1 >= Math.abs(left.layer - right.layer);
    // 计算当前节点高度
    int layer = Math.max(left.layer, right.layer) + 1;
    return new Result(isBalanced, layer);
}

    if (diff == 0 && Objects.isNull(x.left) && Objects.isNull(x.right)) {
        // 如果是叶子节点, 且等于目标和, 就加入到答案中
        ans.add(copy(path));
    }
    if (Objects.nonNull(x.left)) {
        process(x.left, path, diff, ans);
    }
    if (Objects.nonNull(x.right)) {
        process(x.right, path, diff, ans);
    }
    // 结束递归前, 恢复现场.
    path.pollLast();
}

private List<Integer> copy(List<Integer> arr) {
    return new ArrayList<>(arr);
}

实现一个特殊的栈，在基本功能的基础上，再实现返回栈中最小元素的功能

public class MinStack {

    private Deque<Integer> stack = new LinkedList<>();
    private Deque<Integer> minStack = new LinkedList<>();

    /**
     * 压栈
     */
    public void push(int item) {
        stack.push(item);
        Integer min = minStack.peek();
        if (item <= Optional.ofNullable(min).orElse(Integer.MAX_VALUE)) {
            minStack.push(item);
        } else {
            minStack.push(min);
        }
    }

    /**
     * 弹栈
     */
    public int pop() {
        if (stack.isEmpty()) {
            throw new RuntimeException("stack empty...");
        }
        Integer result = stack.pollFirst();
        minStack.pollFirst();
        return result;
    }

    /**
     * 获得栈顶元素, 但不弹栈
     */
    public int peek() {
        if (stack.isEmpty()) {
            throw new RuntimeException("stack empty...");
        }
        return stack.peekFirst();
    }

    /**
     * 获取当前栈中的最小值
     */
    public int getMin() {
        if (minStack.isEmpty()) {
            throw new RuntimeException("stack empty...");
        }
        return minStack.peekFirst();
    }
}

public class StackByQueue {

    private Queue<Integer> X = new LinkedList<>();
    private Queue<Integer> Y = new LinkedList<>();

    /**
     * 压栈
     */
    public void push(int item) {
        if (X.isEmpty() && Y.isEmpty()) {
            X.offer(item);
        } else if (X.isEmpty()) {
            Y.offer(item);
        } else {
            X.offer(item);
        }
    }

    /**
     * 弹栈
     */
    public int pop() {
        if (X.isEmpty() && Y.isEmpty()) {
            throw new RuntimeException("stack empty...");
        }
        Integer result;
        if (X.isEmpty()) {
            while (Y.size() > 1) {
                X.offer(Y.poll());
            }
            result = Y.poll();
        } else {
            while (X.size() > 1) {
                Y.offer(X.poll());
            }
            result = X.poll();
        }
        return result;
    }

    /**
     * 获得栈顶元素, 但不弹栈
     */
    public int peek() {
        if (X.isEmpty() && Y.isEmpty()) {
            throw new RuntimeException("stack empty...");
        }

        if (X.isEmpty()) {

            while (X.size() > 1) {
                Y.offer(X.poll());
            }
            result = X.poll();
            Y.offer(result);
        }
        return result;
    }

/**
 * @author Mr.Lee
 * @description: 用栈实现队列
 * @date 2021-11-15 21:24:36
 */
public class QueueByStack {

    private int size = 0;

    /**
     * 入队时的容器
     */
    private Stack<Integer> in;

    /**
     * 出队时的容器
     */
    private Stack<Integer> out;

    public QueueByStack() {
        in = new Stack<>();
        out = new Stack<>();
    }

    /**
     * 入队
     */
    public void offer(int val) {
        in.push(val);
        size++;
    }

    /**
     * 出队
     */
    public int poll() {
        if (size == 0) {
            throw new RuntimeException("queue empty...");
        }
        if (out.isEmpty()) {
            // 从入队容器里, 拷贝到出队容器中
            while (!in.isEmpty()) {
                out.push(in.pop());
            }
        }
        size--;
        return out.pop();
    }

    /**
     * 队首元素(不出队)
     */
    public int peek() {
        if (size == 0) {
            throw new RuntimeException("queue empty...");
        }
        if (out.isEmpty()) {
            // 从入队容器里, 拷贝到出队容器中
            while (!in.isEmpty()) {
                out.push(in.pop());
            }
        }
        return out.peek();
    }

    public int size() {
        return size;
    }

    public boolean isEmpty() {
        return size == 0;
    }
}

# 1. 时间复杂度分析(Master公式)

用一个简单的递归, 来理解Master公式

/*
 * 递归调用求最大值
 */
public int process(int[] arr, int L, int R) {
    if (L == R) {
        return arr[L];
    }
    int mid = L + ((R - L) >> 1); //求中点
    int leftMax = process(arr, L, mid);
    int rightMax = process(arr, mid + 1, R);
    return Math.max(leftMax, rightMax);
}

解析:

程序进行递归调用是，符合==子问题规模是等规模==（leftMax和rightMax）;

a: 子问题调用的次数，本程序的次数为2次（leftMax和rightMax）；

N：为母问题规模，本程序的母问题规模为N；

T(N/b): 子问题规模，本程序的子问题规模为：N/2（二分）；

O(n^d)： 子问题以为的时间复杂度，本程序为O(1)(求中点)；

可得本程序的master公式为：T(N) = 2 * T(N/2) + O(1);

所以，本程序的时间复杂度为：T(N) = O(N)
