自定义对象, 绑定get, set 方法

自定义对象, 通过 defineSetter 与 defineGetter 绑定get, set 方法

var c = {sum:"", show:function(num, num1) {
    sum = num + num1;
}};

c.__defineSetter__('scum', function(y) {
    this.sum = sum + "..." + y;
});
c.__defineGetter__('scum', function() {
    return "sum = " + sum;
});

c.show(1,3);
c.scum = "1";

alert(c.sum);
alert(c.scum);
